4   import java.net.URI;
5   import java.security.GeneralSecurityException;
6   import java.util.Set;
7   
8   import javax.json.JsonObject;
9   import javax.validation.constraints.NotNull;

11  import net.trajano.openidconnect.auth.AuthenticationRequest;
12  import net.trajano.openidconnect.core.Scope;
13  import net.trajano.openidconnect.token.IdToken;
14  import net.trajano.openidconnect.token.IdTokenResponse;
15  
16  /**
17   * This provides storage and retrieval for the token responses. Implementers
18   * would generally be providing an expiring cache that provides multiple key
19   * types pointing to the same token response instance.
20   *
21   * @author Archimedes
22   */
23  public interface TokenProvider {
24  
25      /**
26       * Gets the token data by access token. This may return null if there is no
27       * data found for the access token.
28       *
29       * @param accessToken
30       *            access token.
31       * @return token response data
32       */
33      IdTokenResponse getByAccessToken(String accessToken);
34  
35      /**
36       * Gets the token by authorization code.
37       *
38       * @param code
39       * @param deleteAfterRetrieval
40       *            if true the implementation is expected to remove the code
41       *            mapping to the token after the call
42       * @return
43       */
44      IdTokenResponse getByCode(String code,
45              boolean deleteAfterRetrieval);
46  
47      /**
48       * @param clientId
49       * @param refreshToken
50       * @param scopes
51       *            this may be null in which case the original scope is used, but
52       *            can be used to reduce the scope after refresh.
53       * @param expiresIn
54       *            time in seconds of how long the new token will expire this
55       *            will also reissue the ID token. This may be null to use the
56       *            provider's default
57       * @return
58       * @throws GeneralSecurityException
59       * @throws IOException
60       */
61      IdTokenResponse refreshToken(@NotNull String clientId,
62              @NotNull String refreshToken,
63              Set<Scope> scopes,
64              Integer expiresIn) throws IOException,
65                      GeneralSecurityException;
66  
67      /**
68       * Builds an {@link IdToken} for the subject . Extra implementation specific
69       * options to the storage process to change certain aspects of the token can
70       * be provided as well.
71       *
72       * @param subject
73       *            subject
74       * @return authorization code
75       */
76      String createNewToken(String subject,
77              URI issuerUri,
78              AuthenticationRequest request) throws IOException,
79                      GeneralSecurityException;
80  
81      IdTokenResponse getByConsent(Consent consent);
82  
83      JsonObject getClaimsByAccessToken(String accessToken);
84  }
