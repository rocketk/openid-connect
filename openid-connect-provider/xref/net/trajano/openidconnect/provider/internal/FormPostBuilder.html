3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.net.URI;
6   import java.util.HashMap;
7   import java.util.Map;
8   import java.util.Scanner;
9   
10  public class FormPostBuilder {
11  
12      private static final String TEMPLATE;
13  
14      private static final String INPUT_HIDDEN_FORMAT = "<input type='hidden' name='%1s' value='%2s' />";
15      static {
16          try {
17              InputStream templateStream = Thread.currentThread()
18                      .getContextClassLoader()
19                      .getResourceAsStream("META-INF/form_post.html");
20              Scanner scanner = new Scanner(templateStream);
21              TEMPLATE = scanner.useDelimiter("\\A")
22                      .next();
23              scanner.close();
24              templateStream.close();
25          } catch (IOException e) {
26              throw new RuntimeException(e);
27          }
28      }
29  
30      public FormPostBuilder(URI redirectUri) {
31  
32          this.redirectUri = redirectUri;
33      }
34  
35      private final URI redirectUri;
36  
37      private final Map<String, String> inputMap = new HashMap<>();
38  
39      public String buildFormPost() {
40  
41          StringBuilder inputs = new StringBuilder();
42          for (Map.Entry<String, String> entry : inputMap.entrySet()) {
43              inputs.append(String.format(INPUT_HIDDEN_FORMAT, entry.getKey(), entry.getValue()));
44          }
45          return String.format(TEMPLATE, redirectUri, inputs);
46      }
47  
48      public void put(String name,
49              String value) {
50  
51              inputMap.put(name, value);
52      }
53  }
