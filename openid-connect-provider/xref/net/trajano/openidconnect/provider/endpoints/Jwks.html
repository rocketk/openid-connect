3   import javax.ejb.EJB;
4   import javax.ws.rs.GET;
5   import javax.ws.rs.Path;
6   import javax.ws.rs.Produces;
7   import javax.ws.rs.core.CacheControl;
8   import javax.ws.rs.core.MediaType;
9   import javax.ws.rs.core.Response;
10  
11  import net.trajano.openidconnect.provider.spi.KeyProvider;
12  
13  /**
14   * <p>
15   * The Authorization Endpoint performs Authentication of the End-User. This is
16   * done by sending the User Agent to the Authorization Server's Authorization
17   * Endpoint for Authentication and Authorization, using request parameters
18   * defined by OAuth 2.0 and additional parameters and parameter values defined
19   * by OpenID Connect.
20   * </p>
21   * <p>
22   * Communication with the Authorization Endpoint MUST utilize TLS. See Section
23   * 16.17 for more information on using TLS.
24   * </p>
25   *
26   * @author Archimedes Trajano
27   */
28  @Path("jwks")
29  public class Jwks {
30  
31      private KeyProvider keyProvider;
32  
33      /**
34       * <p>
35       * An Authentication Request is an OAuth 2.0 Authorization Request that
36       * requests that the End-User be authenticated by the Authorization Server.
37       * </p>
38       * <p>
39       * Authorization Servers MUST support the use of the HTTP GET and POST
40       * methods defined in RFC 2616 [RFC2616] at the Authorization Endpoint.
41       * Clients MAY use the HTTP GET or POST methods to send the Authorization
42       * Request to the Authorization Server. If using the HTTP GET method, the
43       * request parameters are serialized using URI Query String Serialization,
44       * per Section 13.1. If using the HTTP POST method, the request parameters
45       * are serialized using Form Serialization, per Section 13.2.
46       * </p>
47       * <p>
48       * Only 3 keys are provided at a time
49       * </p>
50       */
51      @GET
52      @Produces(MediaType.APPLICATION_JSON)
53      public Response op() {
54  
55          final CacheControl cacheControl = new CacheControl();
56          cacheControl.setPrivate(false);
57          cacheControl.setMaxAge(300);
58  
59          return Response.ok(keyProvider.getJwks())
60                  .cacheControl(cacheControl)
61                  .tag(keyProvider.getSecretKeyId())
62                  .build();
63      }
64  
65      @EJB
66      public void setKeyProvider(final KeyProvider keyProvider) {
67  
68          this.keyProvider = keyProvider;
69      }
70  
71  }
