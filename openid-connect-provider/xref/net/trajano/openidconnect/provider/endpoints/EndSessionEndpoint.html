38  /**
39   * An RP can notify the OP that the End-User has logged out of the site, and
40   * might want to log out of the OP as well. In this case, the RP, after having
41   * logged the End-User out of the RP, redirects the End-User's User Agent to the
42   * OP's logout endpoint URL. This URL is normally obtained via the
43   * end_session_endpoint element of the OP's Discovery response, or may be
44   * learned via other mechanisms.
45   *
46   * @author Archimedes
47   */
48  @Path("end")
49  @Stateless
50  @Produces(MediaType.APPLICATION_JSON)
51  public class EndSessionEndpoint {
52  
53      @EJB
54      private Authenticator authenticator;
55  
56      @EJB
57      private ClientManager clientManager;
58  
59      @EJB
60      private KeyProvider keyProvider;
61  
62      @Context
63      private Providers providers;
64  
65      @EJB
66      private TokenProvider tokenProvider;
67  
68      /**
69       * @param nonce
70       *            nonce
71       * @param logout
72       *            if true then the session is terminated before the redirect
73       * @param req
74       * @return
75       * @throws IOException
76       * @throws GeneralSecurityException
77       */
78      @POST
79      @Path("confirm")
80      @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
81      public Response confirm(@NotNull @FormParam("nonce") final String nonce,
82          @NotNull @FormParam("logout") final boolean logout,
83          @Context final HttpServletRequest req) throws IOException,
84              GeneralSecurityException {
85  
86          final HttpSession session = req.getSession(false);
87          if (session == null || !session.getAttribute("nonce")
88              .equals(nonce)) {
89              throw new OpenIdConnectException(ErrorCode.access_denied);
90          }
91  
92          final URI postLogoutRedirectUri = (URI) session.getAttribute("post_logout_redirect_uri");
93          final String state = (String) session.getAttribute("state");
94          if (logout) {
95              authenticator.endSession(req);
96              session.invalidate();
97          }
98          return Response.temporaryRedirect(UriBuilder.fromUri(postLogoutRedirectUri)
99              .queryParam(OpenIdConnectKey.STATE, state)
100             .build())
101             .build();
102 
103     }

105     /**
106      * <p>
107      * The Client sends the UserInfo Request using either HTTP GET or HTTP POST.
108      * The Access Token obtained from an OpenID Connect Authentication Request
109      * MUST be sent as a Bearer Token, per Section 2 of OAuth 2.0 Bearer Token
110      * Usage [RFC6750].
111      * </p>
112      * <p>
113      * It is RECOMMENDED that the request use the HTTP GET method and the Access
114      * Token be sent using the Authorization header field.
115      * </p>
116      *
117      * @param req
118      * @return
119      * @throws GeneralSecurityException
120      * @throws IOException
121      */
122     @GET
123     public Response getOp(@QueryParam("post_logout_redirect_uri") final URI postLogoutRedirectUri,
124         @QueryParam(OpenIdConnectKey.ID_TOKEN_HINT) final String idTokenHint,
125         @QueryParam(OpenIdConnectKey.STATE) final String state,
126         @Context final HttpServletRequest req) throws IOException,
127             GeneralSecurityException {
128 
129         return op(postLogoutRedirectUri, idTokenHint, state, req);
130     }

132     /**
133      * <p>
134      * This endpoint will store the needed logout information in the HTTP
135      * session when available.
136      * </p>
137      * <p>
138      * The <code>id_token_hint</code> is required by this implementation.
139      * </p>
140      *
141      * @param postLogoutRedirectUri
142      *            OPTIONAL. URL to which the RP is requesting that the
143      *            End-User's User Agent be redirected after a logout has been
144      *            performed. The value MUST have been previously registered with
145      *            the OP, either using the post_logout_redirect_uris
146      *            Registration parameter or via another mechanism. If supplied,
147      *            the OP SHOULD honor this request following the logout.
148      * @param idTokenHint
149      *            RECOMMENDED. Previously issued ID Token passed to the logout
150      *            endpoint as a hint about the End-User's current authenticated
151      *            session with the Client. This is used as an indication of the
152      *            identity of the End-User that the RP is requesting be logged
153      *            out by the OP. The OP need not be listed as an audience of the
154      *            ID Token when it is used as an id_token_hint value.
155      * @param state
156      *            OPTIONAL. Opaque value used by the RP to maintain state
157      *            between the logout request and the callback to the endpoint
158      *            specified by the post_logout_redirect_uri parameter. If
159      *            included in the logout request, the OP passes this value back
160      *            to the RP using the state query parameter when redirecting the
161      *            User Agent back to the RP.
162      * @param req
163      * @return
164      * @throws GeneralSecurityException
165      * @throws IOException
166      */
167     @POST
168     public Response op(@FormParam("post_logout_redirect_uri") final URI postLogoutRedirectUri,
169         @FormParam(OpenIdConnectKey.ID_TOKEN_HINT) final String idTokenHint,
170         @FormParam(OpenIdConnectKey.STATE) final String state,
171         @Context final HttpServletRequest req) throws IOException,
172             GeneralSecurityException {
173 
174         IdToken idToken = null;
175         if (idTokenHint != null) {
176             final JsonWebTokenProcessor idTokenProcessor = new JsonWebTokenProcessor(idTokenHint).jwks(keyProvider.getPrivateJwks());
177             if (!idTokenProcessor.isJwkAvailable()) {
178                 throw new OpenIdConnectException(ErrorCode.invalid_request, "no jwk available for kid");
179             }
180             final byte[] idTokenBytes = idTokenProcessor.getPayload();
181             final MessageBodyReader<IdToken> idTokenReader = providers.getMessageBodyReader(IdToken.class, IdToken.class, new Annotation[0], MediaType.APPLICATION_JSON_TYPE);
182             idToken = idTokenReader.readFrom(IdToken.class, IdToken.class, new Annotation[0], MediaType.APPLICATION_JSON_TYPE, null, new ByteArrayInputStream(idTokenBytes));
183         }
184         if (postLogoutRedirectUri != null && idToken != null && !clientManager.isPostLogoutRedirectUriValidForClient(idToken.getAzp(), postLogoutRedirectUri)) {
185             throw new OpenIdConnectException(ErrorCode.invalid_request);
186         }
187 
188         final HttpSession session = req.getSession(false);
189         if (session != null && authenticator.isAuthenticated(req)) {
190             if (!authenticator.getSubject(req)
191                 .equals(idToken.getSub())) {
192                 throw new OpenIdConnectException(ErrorCode.access_denied);
193             }
194             final UriBuilder uriBuilder = UriBuilder.fromUri(req.getRequestURL()
195                 .toString())
196                 .replacePath(req.getContextPath());
197 
198             session.setAttribute("post_logout_redirect_uri", postLogoutRedirectUri);
199             if (idToken != null) {
200                 session.setAttribute("id_token", idToken);
201             }
202             session.setAttribute("state", state);
203             final String nonce = keyProvider.nextEncodedToken();
204             session.setAttribute("nonce", nonce);
205 
206             return Response.temporaryRedirect(authenticator.logout(nonce, idToken, state, postLogoutRedirectUri, req, uriBuilder))
207                 .build();
208         }
209         // If the user is not authenticated, it just goes back to the post
210         // redirect URI.
211         return Response.temporaryRedirect(UriBuilder.fromUri(postLogoutRedirectUri)
212             .queryParam(OpenIdConnectKey.STATE, state)
213             .build())
214             .build();
215 
216     }
