28  @Path("openid-configuration")
29  public class WellKnownOpenIdConfiguration {

32  
33      public static final String CODE_ID_TOKEN = "code id_token";
34  
35      public static final String CODE_ID_TOKEN_TOKEN = "code id_token token";
36  
37      public static final String CODE_TOKEN = "code token";
38  
39      public static final String ID_TOKEN = "id_token";
40  
41      public static final String ID_TOKEN_TOKEN = "id_token token";

44       * Authorization endpoint mapping that is built during {@link #init()}

46      private String authorizationMapping;

48      /**
49       * JWKS URI mapping that is built during {@link #init()}
50       */
51      private String jwksMapping;

58      private String revocationMapping;
59  
60      /**
61       * Token endpoint mapping that is built during {@link #init()}

63      private String tokenMapping;

66       * Userinfo endpoint mapping that is built during {@link #init()}

70      /**
71       * Determines the endpoints for OAuth2 based on the mappings specified in
72       * the {@link ServletRegistration}s.
73       */
74      @PostConstruct
75      public void init() {
76  
77          // eventually allow for multiple for now hard code.
78          String applicationPath = "V1";
79          if (!applicationPath.startsWith("/")) {
80              applicationPath = "/" + applicationPath;
81          }
82          jwksMapping = applicationPath + "/jwks";
83          authorizationMapping = applicationPath + "/auth";
84          tokenMapping = applicationPath + "/token";
85          userinfoMapping = applicationPath + "/profile";
86          revocationMapping = applicationPath + "/revocation";
87          endSessionMapping = applicationPath + "/end";
88  
89      }
90  
91      private String endSessionMapping;
92  
93      private UserinfoProvider userinfoProvider;
94  
95      @GET
96      @Produces(MediaType.APPLICATION_JSON)
97      public Response op(@Context final HttpServletRequest request) {
98  
99          final OpenIdProviderConfiguration openIdConfiguration = new OpenIdProviderConfiguration();
100 
101         final UriBuilder baseUri = UriBuilder.fromUri(create(request.getRequestURL()
102                 .toString()))
103                 .scheme("https")
104                 .replaceQuery(null)
105                 .fragment(null);
106         openIdConfiguration.setIssuer(baseUri.replacePath(request.getContextPath())
107                 .build());
108         openIdConfiguration.setJwksUri(baseUri.replacePath(request.getContextPath() + jwksMapping)
109                 .build());
110         openIdConfiguration.setAuthorizationEndpoint(baseUri.replacePath(request.getContextPath() + authorizationMapping)
111                 .build());
112         openIdConfiguration.setRevocationEndpoint(baseUri.replacePath(request.getContextPath() + revocationMapping)
113                 .build());
114         openIdConfiguration.setTokenEndpoint(baseUri.replacePath(request.getContextPath() + tokenMapping)
115                 .build());
116         openIdConfiguration.setEndSessionEndpoint(baseUri.replacePath(request.getContextPath() + endSessionMapping)
117                 .build());
118         openIdConfiguration.setUserinfoEndpoint(baseUri.replacePath(request.getContextPath() + userinfoMapping)
119                 .build());
120 
121         final Scope[] scopesSupported = userinfoProvider.scopesSupported();
122         final Scope[] scopes = new Scope[scopesSupported.length + 1];
123         System.arraycopy(scopesSupported, 0, scopes, 1, scopesSupported.length);
124         scopes[0] = Scope.openid;
125         openIdConfiguration.setScopesSupported(scopes);
126 
127         final String[] claimsSupported = userinfoProvider.claimsSupported();
128         final String[] claims = new String[claimsSupported.length + 3];
129         claims[0] = "sub";
130         claims[1] = "iss";
131         claims[2] = "auth_time";
132         System.arraycopy(claimsSupported, 0, claims, 3, claimsSupported.length);
133         openIdConfiguration.setClaimsSupported(claims);
134 
135         openIdConfiguration.setResponseTypesSupported(CODE, ID_TOKEN, ID_TOKEN_TOKEN, CODE_ID_TOKEN, CODE_TOKEN, CODE_ID_TOKEN_TOKEN);
136         openIdConfiguration.setRequestParameterSupported(true);
137         openIdConfiguration.setGrantTypesSupported(GrantType.authorization_code, GrantType.implicit);
138         openIdConfiguration.setRequestUriParameterSupported(false);
139         openIdConfiguration.setSubjectTypesSupported(SubjectIdentifierType.PUBLIC);
140         openIdConfiguration.setTokenEndpointAuthMethodsSupported(TokenEndPointAuthMethod.client_secret_basic, TokenEndPointAuthMethod.client_secret_post);
141         openIdConfiguration.setIdTokenSigningAlgValuesSupported(JsonWebAlgorithm.getSigAlgorithms());
142         openIdConfiguration.setRequestObjectEncryptionAlgValuesSupported(JsonWebAlgorithm.getKexAlgorithms());
143         openIdConfiguration.setRequestObjectEncryptionEncValuesSupported(JsonWebAlgorithm.getEncAlgorithms());
144         openIdConfiguration.setResponseModesSupported(ResponseMode.fragment, ResponseMode.query, ResponseMode.form_post);
145 
146         final CacheControl cacheControl = new CacheControl();
147         cacheControl.setPrivate(false);
148         cacheControl.setMaxAge(86400);
149 
150         return Response.ok(openIdConfiguration)
151                 .cacheControl(cacheControl)
152                 .tag(keyProvider.getSecretKeyId())
153                 .build();
154 
155     }
