37  /**
38   * <p>
39   * The Authorization Endpoint performs Authentication of the End-User. This is
40   * done by sending the User Agent to the Authorization Server's Authorization
41   * Endpoint for Authentication and Authorization, using request parameters
42   * defined by OAuth 2.0 and additional parameters and parameter values defined
43   * by OpenID Connect.
44   * </p>
45   * <p>
46   * Communication with the Authorization Endpoint MUST utilize TLS. See Section
47   * 16.17 for more information on using TLS.
48   * </p>
49   *
50   * @author Archimedes Trajano
51   */

86      /**
87       * <p>
88       * An Authentication Request is an OAuth 2.0 Authorization Request that
89       * requests that the End-User be authenticated by the Authorization Server.
90       * </p>
91       * <p>
92       * Authorization Servers MUST support the use of the HTTP GET and POST
93       * methods defined in RFC 2616 [RFC2616] at the Authorization Endpoint.
94       * Clients MAY use the HTTP GET or POST methods to send the Authorization
95       * Request to the Authorization Server. If using the HTTP GET method, the
96       * request parameters are serialized using URI Query String Serialization,
97       * per Section 13.1. If using the HTTP POST method, the request parameters
98       * are serialized using Form Serialization, per Section 13.2.
99       * </p>
100      *
101      * @return JAX-RS Response
102      * @throws GeneralSecurityException
103      * @throws IOException
104      */
105     @POST
106     @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
107     public Response op(@Context final HttpServletRequest req) throws IOException,
108         GeneralSecurityException {
109 
110         final AuthenticationRequest authenticationRequest = new AuthenticationRequest(req, keyProvider.getPrivateJwks());
111 
112         if (!clientManager.isRedirectUriValidForClient(authenticationRequest.getClientId(), authenticationRequest.getRedirectUri())) {
113             throw new OpenIdConnectException(invalid_grant, "redirect URI is not supported for the client");
114         }
115 
116         final boolean authenticated = authenticator.isAuthenticated(req);
117 
118         if (!authenticated && authenticationRequest.getPrompts()
119             .contains(Prompt.none)) {
120             throw new RedirectedOpenIdProviderException(authenticationRequest, new ErrorResponse(login_required));
121         }
122 
123         final Consent consent = new Consent(authenticator.getSubject(req), authenticationRequest.getClientId(), authenticationRequest.getScopes());
124 
125         final boolean consented = clientManager.isImplicitConsent(authenticationRequest.getClientId()) || tp.getByConsent(consent) != null;
126         if (!consented && authenticationRequest.getPrompts()
127             .contains(Prompt.none)) {
128             throw new RedirectedOpenIdProviderException(authenticationRequest, new ErrorResponse(consent_required));
129         }
130 
131         String reqJwt = req.getParameter(OpenIdConnectKey.REQUEST);
132         if (reqJwt == null) {
133             final JsonWebTokenBuilder b = new JsonWebTokenBuilder().payload(authenticationRequest.toJsonObject())
134                 .compress(true);
135             reqJwt = b.toString();
136         }
137 
138         final UriBuilder contextUriBuilder = UriBuilder.fromPath(req.getServletContext().getContextPath() + "/");
139         if (!authenticated) {
140 
141             // Workaround for WebSphere Liberty as Response.temporaryRedirect will treat URIs that do not have a scheme as relative.
142             final URI targetUri = authenticator.authenticate(authenticationRequest, reqJwt, req, contextUriBuilder);
143             return Response.temporaryRedirect(URI.create(req.getRequestURL().toString()).resolve(targetUri))
144                 .build();
145 
146         } else if (!consented) {
147 
148             final URI targetUri = authenticator.consent(authenticationRequest, reqJwt, req, contextUriBuilder);
149             return Response.temporaryRedirect(URI.create(req.getRequestURL().toString()).resolve(targetUri))
150                 .build();
151         } else {
152 
153             return arp.buildResponse(reqJwt, req, authenticator.getSubject(req));
154         }
155 
156     }
