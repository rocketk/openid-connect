69      @Test
70      public void testPublicPrivateJwksToJsonEquivalency() throws Exception {
71  
72          final DefaultKeyProvider keyProvider = new DefaultKeyProvider();
73          keyProvider.generateKeys();
74  
75          final MessageBodyWriter<JsonWebKeySet> writer = providers.getMessageBodyWriter(JsonWebKeySet.class, null, null, MediaType.APPLICATION_JSON_TYPE);
76  
77          final JsonArray privateJwks;
78          {
79              final ByteArrayOutputStream baos = new ByteArrayOutputStream();
80              writer.writeTo(keyProvider.getPrivateJwks(), JsonWebKeySet.class, null, null, MediaType.APPLICATION_JSON_TYPE, null, baos);
81              baos.close();
82              final JsonReader jsonReader = Json.createReader(new ByteArrayInputStream(baos.toByteArray()));
83              privateJwks = jsonReader.readObject()
84                      .getJsonArray("keys");
85          }
86          final JsonArray publicJwks;
87          {
88              final ByteArrayOutputStream baos = new ByteArrayOutputStream();
89              writer.writeTo(keyProvider.getJwks(), JsonWebKeySet.class, null, null, MediaType.APPLICATION_JSON_TYPE, null, baos);
90              baos.close();
91              final JsonReader jsonReader = Json.createReader(new ByteArrayInputStream(baos.toByteArray()));
92              publicJwks = jsonReader.readObject()
93                      .getJsonArray("keys");
94          }
95  
96          int found = 0;
97          for (int i = 0; i < publicJwks.size(); ++i) {
98              for (int j = 0; j < privateJwks.size(); ++j) {
99                  final JsonObject publicJwk = publicJwks.getJsonObject(i);
100                 final JsonObject privateJwk = privateJwks.getJsonObject(j);
101                 if (publicJwk.getString("kid")
102                         .equals(privateJwk.getString("kid"))) {
103                     assertEquals(publicJwk.getString("kid"), privateJwk.getString("kid"));
104                     assertEquals(publicJwk.getString("alg"), privateJwk.getString("alg"));
105                     assertEquals("sig", publicJwk.getString("use"));
106                     assertEquals("enc", privateJwk.getString("use"));
107                     assertEquals(publicJwk.getString("n"), privateJwk.getString("n"));
108                     found++;
109                 }
110             }
111         }
112         assertEquals(publicJwks.size(), found);
113 
114     }
