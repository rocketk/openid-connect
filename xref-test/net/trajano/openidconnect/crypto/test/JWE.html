3   import java.io.IOException;
4   import java.security.GeneralSecurityException;
5   import java.util.zip.DataFormatException;
6   
7   import javax.json.JsonObject;
8   
9   import net.trajano.openidconnect.crypto.JsonWebKey;
10  import net.trajano.openidconnect.crypto.JsonWebToken;
11  import net.trajano.openidconnect.crypto.JsonWebTokenBuilder;
12  import net.trajano.openidconnect.crypto.JsonWebTokenProcessor;
13  import net.trajano.openidconnect.internal.CharSets;
14  
15  /**
16   * Support class used for testing JWE.
17   * 
18   * @author Archimedes
19   */
20  public class JWE {
21  
22      public static byte[] decrypt(final JsonWebToken jsonWebToken,
23              final JsonWebKey jwk) throws IOException,
24              GeneralSecurityException {
25  
26          JsonWebTokenProcessor p = new JsonWebTokenProcessor(jsonWebToken);
27          p.allowJwkToBeSet(true)
28                  .jwk(jwk);
29          return p.getPayload();
30      }
31  
32      public static byte[] decrypt(final String jwe,
33              final JsonWebKey jwk) throws IOException,
34              GeneralSecurityException,
35              DataFormatException {
36  
37          final JsonWebToken jsonWebToken = new JsonWebToken(jwe);
38          return decrypt(jsonWebToken, jwk);
39      }
40  
41      public static String encrypt(final byte[] plaintext,
42              final JsonWebKey jwk,
43              final String alg,
44              final String enc) throws IOException,
45              GeneralSecurityException {
46  
47          return encrypt(plaintext, jwk, alg, enc, false);
48      }
49  
50      public static String encrypt(final byte[] plaintext,
51              final JsonWebKey jwk,
52              final String alg,
53              final String enc,
54              final boolean compress) throws IOException,
55              GeneralSecurityException {
56  
57          JsonWebTokenBuilder b = new JsonWebTokenBuilder();
58          b.payload(plaintext);
59          b.jwk(jwk);
60          b.alg(alg);
61          b.enc(enc);
62          b.compress(compress);
63          return b.toString();
64      }
65  
66      public static String encrypt(final JsonObject obj,
67              final JsonWebKey jwk,
68              final String alg,
69              final String enc) throws IOException,
70              GeneralSecurityException {
71  
72          return encrypt(obj.toString()
73                  .getBytes(CharSets.UTF8), jwk, alg, enc, false);
74      }
75  }
