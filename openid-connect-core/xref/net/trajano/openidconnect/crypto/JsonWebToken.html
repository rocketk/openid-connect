5   /**
6    * The JSON Web Token. It is comprised of a header that is a Base64url encoded
7    * JSON followed by 1 to many Base64url encoded payloads joined by '.'
8    * character. This class is immutable
9    *
10   * @author Archimedes
11   */
12  public class JsonWebToken {
13  
14      /**
15       * Constant for the "none" algorithm.
16       */
17      public static final String ALG_NONE = "none";
18  
19      /**
20       * <p>
21       * "alg" (Algorithm) Header Parameter
22       * </p>
23       * <p>
24       * This parameter has the same meaning, syntax, and processing rules as the
25       * alg Header Parameter defined in Section 4.1.1 of [JWS], except that the
26       * Header Parameter identifies the cryptographic algorithm used to encrypt
27       * or determine the value of the Content Encryption Key (CEK). The encrypted
28       * content is not usable if the alg value does not represent a supported
29       * algorithm, or if the recipient does not have a key that can be used with
30       * that algorithm.
31       * </p>
32       * <p>
33       * A list of defined alg values for this use can be found in the IANA JSON
34       * Web Signature and Encryption Algorithms registry defined in [JWA]; the
35       * initial contents of this registry are the values defined in Section 4.1
36       * of the JSON Web Algorithms (JWA) [JWA] specification.
37       * </p>
38       */
39      private final String alg;
40  
41      /**
42       * <p>
43       * "enc" (Encryption Algorithm) Header Parameter
44       * </p>
45       * <p>
46       * The enc (encryption algorithm) Header Parameter identifies the content
47       * encryption algorithm used to perform authenticated encryption on the
48       * Plaintext to produce the Ciphertext and the Authentication Tag. This
49       * algorithm MUST be an AEAD algorithm with a specified key length. The
50       * encrypted content is not usable if the enc value does not represent a
51       * supported algorithm. enc values should either be registered in the IANA
52       * JSON Web Signature and Encryption Algorithms registry defined in [JWA] or
53       * be a value that contains a Collision-Resistant Name. The enc value is a
54       * case-sensitive ASCII string containing a StringOrURI value. This Header
55       * Parameter MUST be present and MUST be understood and processed by
56       * implementations.
57       * </p>
58       * <p>
59       * A list of defined enc values for this use can be found in the IANA JSON
60       * Web Signature and Encryption Algorithms registry defined in [JWA]; the
61       * initial contents of this registry are the values defined in Section 5.1
62       * of the JSON Web Algorithms (JWA) [JWA] specification.
63       * </p>
64       */
65      private final String enc;
66  
67      private final String joseHeaderEncoded;
68  
69      private final byte[][] payloads;
70  
71      /**
72       * <p>
73       * "zip" (Compression Algorithm) Header Parameter
74       * <p>
75       * <p>
76       * The zip (compression algorithm) applied to the Plaintext before
77       * encryption, if any. The zip value defined by this specification is:
78       * </p>
79       * <ul>
80       * <li>DEF - Compression with the DEFLATE [RFC1951] algorithm
81       * </ul>
82       * <p>
83       * Other values MAY be used. Compression algorithm values can be registered
84       * in the IANA JSON Web Encryption Compression Algorithm registry defined in
85       * [JWA]. The zip value is a case-sensitive string. If no zip parameter is
86       * present, no compression is applied to the Plaintext before encryption.
87       * When used, this Header Parameter MUST be integrity protected; therefore,
88       * it MUST occur only within the JWE Protected Header. Use of this Header
89       * Parameter is OPTIONAL. This Header Parameter MUST be understood and
90       * processed by implementations.
91       * </p>
92       */
93      private final String zip;
94  
95      /**
96       * "kid" (Key ID) Header Parameter
97       * <p>
98       * This parameter has the same meaning, syntax, and processing rules as the
99       * kid Header Parameter defined in Section 4.1.4 of [JWS], except that the
100      * key hint references the public key to which the JWE was encrypted; this
101      * can be used to determine the private key needed to decrypt the JWE. This
102      * parameter allows originators to explicitly signal a change of key to JWE
103      * recipients.
104      */
105     private final String kid;
106 
107     public JsonWebToken(final JoseHeader joseHeader, final byte[][] payloads) {
108 
109         joseHeaderEncoded = new String(joseHeader.getEncoded());
110         alg = joseHeader.getAlg();
111         enc = joseHeader.getEnc();
112         kid = joseHeader.getKid();
113         zip = joseHeader.getZip();
114         this.payloads = payloads;
115     }
116 
117     public JsonWebToken(final String jwt) throws IOException {
118 
119         final String[] tokens = jwt.split("\\.");
120 
121         joseHeaderEncoded = tokens[0];
122         final JoseHeader joseHeader = new JoseHeader(Encoding.base64urlDecodeToString(joseHeaderEncoded));
123 
124         alg = joseHeader.getAlg();
125         enc = joseHeader.getEnc();
126         kid = joseHeader.getKid();
127         zip = joseHeader.getZip();
128 
129         payloads = new byte[tokens.length - 1][];
130         for (int i = 1; i < tokens.length; ++i) {
131             payloads[i - 1] = Encoding.base64urlDecode(tokens[i]);
132         }
133 
134     }
135

151     /**
152      * This retrieves the JOSE header. Please note that this is an expensive
153      * process because it will perform decoding of the data. The JOSE header
154      * returned may be mutable but it is not associated with this object.
155      *
156      * @return JOSE header
157      */
158     public JoseHeader getJoseHeader() {
159 
160         return new JoseHeader(Encoding.base64urlDecodeToString(joseHeaderEncoded));
161     }
162 
163     /**
164      * Gets the encoded JOSE Header as it was provided.
165      *
166      * @return encoded JOSE header.
167      */
168     public String getJoseHeaderEncoded() {
169 
170         return joseHeaderEncoded;
171     }
172 
173     public int getNumberOfPayloads() {
174 
175         return payloads.length;
176     }
177 
178     public byte[] getPayload(final int i) {
179 
180         return payloads[i];
181     }
182 
183     public String getZip() {
184 
185         return zip;
186     }
187 
188     /**
189      * Builds the serialized JWT.
190      */
191     @Override
192     public String toString() {
193 
194         final StringBuilder b = new StringBuilder(joseHeaderEncoded);
195         for (final byte[] payload : payloads) {
196             b.append('.')
197                     .append(Encoding.base64urlEncode(payload));
198         }
199         return b.toString();
200 
201     }
