16  /**
17   * Renders the {@link OpenIdConnectException} as a redirect with an optional
18   * state value. It will only render {@value } OpenIdConnectKey#ERROR},
19   * {@value OpenIdConnectKey#ERROR_DESCRIPTION} and {@value }
20   * {@value OpenIdConnectKey#ERROR_URI} from the response.
21   * 
22   * @author Archimedes Trajano
23   */
24  public class RedirectedOpenIdProviderException extends WebApplicationException {
25  
26      /**
27       *
28       */
29      private static final long serialVersionUID = -8684305650194342408L;
30  
31      private static Response responseBuilder(final URI redirectUri,
32          final ErrorResponse errorResponse,
33          final String state,
34          final boolean useFragment) {
35  
36          UriBuilder b = UriBuilder.fromUri(redirectUri)
37              .queryParam(ERROR, errorResponse.getError());
38          if (errorResponse.getErrorDescription() != null) {
39              b.queryParam(ERROR_DESCRIPTION, errorResponse.getErrorDescription());
40          }
41          if (errorResponse.getErrorUri() != null) {
42              b.queryParam(ERROR_URI, errorResponse.getErrorUri());
43          }
44          if (state != null) {
45              b.queryParam(STATE, state);
46          }
47          if (useFragment) {
48              b = UriBuilder.fromUri(redirectUri).fragment(b.build().getQuery());
49  
50          }
51          return Response.temporaryRedirect(b.build())
52              .build();
53  
54      }
55  
56      /**
57       * Convenience constructor as this exception type is generally part of
58       * authentication.
59       * 
60       * @param authenticationRequest
61       * @param errorResponse
62       */
63      public RedirectedOpenIdProviderException(final AuthenticationRequest authenticationRequest, ErrorResponse errorResponse) {
64  
65          this(authenticationRequest, errorResponse, !authenticationRequest.isAuthorizationCodeFlow());
66      }
67  
68      public RedirectedOpenIdProviderException(final AuthenticationRequest authenticationRequest, ErrorResponse errorResponse, boolean useFragment) {
69  
70          this(authenticationRequest.getRedirectUri(), errorResponse, authenticationRequest.getState(), useFragment);
71      }
72  
73      public RedirectedOpenIdProviderException(URI redirectUri, ErrorResponse errorResponse) {
74  
75          this(redirectUri, errorResponse, null);
76      }
77  
78      public RedirectedOpenIdProviderException(URI redirectUri, ErrorResponse errorResponse, String state) {
79  
80          super(responseBuilder(redirectUri, errorResponse, state, false));
81      }
82  
83      public RedirectedOpenIdProviderException(URI redirectUri, ErrorResponse errorResponse, String state, boolean useFragment) {
84  
85          super(responseBuilder(redirectUri, errorResponse, state, useFragment));
86      }
87  }
