32  // TODO should we have the reader in jaspic and the writer in the rest api?
33  @Provider
34  @Produces(MediaType.APPLICATION_JSON)
35  @Consumes(MediaType.APPLICATION_JSON)
36  public class JsonWebKeySetProvider implements MessageBodyReader<JsonWebKeySet>, MessageBodyWriter<JsonWebKeySet> {
37  
38      private Providers providers;
39  
40      @Override
41      public long getSize(final JsonWebKeySet jwks,
42              final Class<?> type,
43              final Type genericType,
44              final Annotation[] annotations,
45              final MediaType mediaType) {
46  
47          return -1;
48      }
49  
50      @Override
51      public boolean isReadable(final Class<?> type,
52              final Type genericType,
53              final Annotation[] annotations,
54              final MediaType mediaType) {
55  
56          return JsonWebKeySet.class.isAssignableFrom(type) && MediaType.APPLICATION_JSON_TYPE.isCompatible(mediaType);
57      }
58  
59      @Override
60      public boolean isWriteable(final Class<?> type,
61              final Type genericType,
62              final Annotation[] annotations,
63              final MediaType mediaType) {
64  
65          return isReadable(type, genericType, annotations, mediaType);
66      }
67  
68      @Override
69      public JsonWebKeySet readFrom(final Class<JsonWebKeySet> type,
70              final Type genericType,
71              final Annotation[] annotations,
72              final MediaType mediaType,
73              final MultivaluedMap<String, String> httpHeaders,
74              final InputStream inputStream) throws IOException,
75              WebApplicationException {
76  
77          final JsonArray keysArray = Json.createReader(inputStream)
78                  .readObject()
79                  .getJsonArray("keys");
80  
81          final JsonWebKeySet keySet = new JsonWebKeySet();
82          final MessageBodyReader<JsonWebKey> reader = providers.getMessageBodyReader(JsonWebKey.class, JsonWebKey.class, annotations, mediaType);
83          for (final JsonValue key : keysArray) {
84              final InputStream keyStream = new ByteArrayInputStream(key.toString()
85                      .getBytes(CharSets.UTF8));
86              final JsonWebKey jsonWebKey = reader.readFrom(JsonWebKey.class, null, annotations, mediaType, null, keyStream);
87              keySet.add(jsonWebKey);
88          }
89  
90          return keySet;
91      }
92  
93      @Context
94      public void setProviders(final Providers providers) {
95  
96          this.providers = providers;
97      }
98  
99      @Override
100     public void writeTo(final JsonWebKeySet jwks,
101             final Class<?> type,
102             final Type genericType,
103             final Annotation[] annotations,
104             final MediaType mediaType,
105             final MultivaluedMap<String, Object> httpHeaders,
106             final OutputStream os) throws IOException,
107             WebApplicationException {
108 
109         final MessageBodyWriter<JsonWebKey> writer = providers.getMessageBodyWriter(JsonWebKey.class, JsonWebKey.class, annotations, mediaType);
110         final JsonArrayBuilder keysArray = Json.createArrayBuilder();
111         for (final JsonWebKey key : jwks.getKeys()) {
112             final ByteArrayOutputStream keyStream = new ByteArrayOutputStream();
113             writer.writeTo(key, JsonWebKey.class, null, annotations, mediaType, null, keyStream);
114             keyStream.close();
115             keysArray.add(Json.createReader(new ByteArrayInputStream(keyStream.toByteArray()))
116                     .readObject());
117         }
118         final JsonObject jwksObject = Json.createObjectBuilder()
119                 .add("keys", keysArray)
120                 .build();
121         final JsonWriter w = Json.createWriter(os);
122         w.write(jwksObject);
123         w.close();
124     }
125 }
