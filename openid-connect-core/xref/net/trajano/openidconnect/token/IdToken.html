13  /**
14   * <p>
15   * <a href="http://openid.net/specs/openid-connect-core-1_0.html#IDToken">ID
16   * Token</a>. The primary extension that OpenID Connect makes to OAuth 2.0 to
17   * enable End-Users to be Authenticated is the ID Token data structure. The ID
18   * Token is a security token that contains Claims about the Authentication of an
19   * End-User by an Authorization Server when using a Client, and potentially
20   * other requested Claims. The ID Token is represented as a JSON Web Token
21   * (JWT).
22   * </p>
23   * <p>
24   * ID Tokens MAY contain other Claims. Any Claims used that are not understood
25   * MUST be ignored. See Sections 3.1.3.6, 3.3.2.11, 5.1, and 7.4 for additional
26   * Claims defined by this specification.
27   * </p>
28   * <p>
29   * ID Tokens MUST be signed using JWS [JWS] and optionally both signed and then
30   * encrypted using JWS [JWS] and JWE [JWE] respectively, thereby providing
31   * authentication, integrity, non-repudiation, and optionally, confidentiality,
32   * per Section 16.14. If the ID Token is encrypted, it MUST be signed then
33   * encrypted, with the result being a Nested JWT, as defined in [JWT]. ID Tokens
34   * MUST NOT use none as the alg value unless the Response Type used returns no
35   * ID Token from the Authorization Endpoint (such as when using the
36   * Authorization Code Flow) and the Client explicitly requested the use of none
37   * at Registration time.
38   * </p>
39   * <p>
40   * ID Tokens SHOULD NOT use the JWS or JWE x5u, x5c, jku, or jwk Header
41   * Parameter fields. Instead, references to keys used are communicated in
42   * advance using Discovery and Registration parameters, per Section 10.
43   * </p>
44   * <p>
45   * This extends {@link Userinfo} as per
46   * http://openid.net/specs/openid-connect-core-1_0.html#StandardClaims which
47   * indicates that the claims can be stored in the token as well.
48   * </p>
49   * 
50   * @author Archimedes Token
51   */
52  @XmlRootElement
53  @XmlAccessorType(XmlAccessType.FIELD)
54  public class IdToken extends Userinfo implements Serializable {
55  
56      /**
57       *
58       */
59      private static final long serialVersionUID = -6920223034971350418L;

61      /**
62       * OPTIONAL. Authentication Context Class Reference. String specifying an
63       * Authentication Context Class Reference value that identifies the
64       * Authentication Context Class that the authentication performed satisfied.
65       * The value "0" indicates the End-User authentication did not meet the
66       * requirements of ISO/IEC 29115 [ISO29115] level 1. Authentication using a
67       * long-lived browser cookie, for instance, is one example where the use of
68       * "level 0" is appropriate. Authentications with level 0 SHOULD NOT be used
69       * to authorize access to any resource of any monetary value. (This
70       * corresponds to the OpenID 2.0 PAPE [OpenID.PAPE] nist_auth_level 0.) An
71       * absolute URI or an RFC 6711 [RFC6711] registered name SHOULD be used as
72       * the acr value; registered names MUST NOT be used with a different meaning
73       * than that which is registered. Parties using this claim will need to
74       * agree upon the meanings of the values used, which may be
75       * context-specific. The acr value is a case sensitive string.
76       */
77      private String acr;
78  
79      /**
80       * OPTIONAL. Authentication Methods References. JSON array of strings that
81       * are identifiers for authentication methods used in the authentication.
82       * For instance, values might indicate that both password and OTP
83       * authentication methods were used. The definition of particular values to
84       * be used in the amr Claim is beyond the scope of this specification.
85       * Parties using this claim will need to agree upon the meanings of the
86       * values used, which may be context-specific. The amr value is an array of
87       * case sensitive strings.
88       */
89      private String[] amr;
90  
91      @XmlElement(name = "at_hash")
92      private String atHash;
93  
94      @XmlElement(name = "c_hash")
95      private String cHash;
96  
97      /**
98       * REQUIRED. Audience(s) that this ID Token is intended for. It MUST contain
99       * the OAuth 2.0 client_id of the Relying Party as an audience value. It MAY
100      * also contain identifiers for other audiences. In the general case, the
101      * aud value is an array of case sensitive strings. In the common special
102      * case when there is one audience, the aud value MAY be a single case
103      * sensitive string.
104      */
105     @XmlElement(required = true)
106     private String aud;
107 
108     /**
109      * Time when the End-User authentication occurred. Its value is a JSON
110      * number representing the number of seconds from 1970-01-01T0:0:0Z as
111      * measured in UTC until the date/time. When a max_age request is made or
112      * when auth_time is requested as an Essential Claim, then this Claim is
113      * REQUIRED; otherwise, its inclusion is OPTIONAL. (The auth_time Claim
114      * semantically corresponds to the OpenID 2.0 PAPE [OpenID.PAPE] auth_time
115      * response parameter.)
116      */
117     @XmlElement(name = "auth_time")
118     private long authTime;
119 
120     /**
121      * OPTIONAL. Authorized party - the party to which the ID Token was issued.
122      * If present, it MUST contain the OAuth 2.0 Client ID of this party. This
123      * Claim is only needed when the ID Token has a single audience value and
124      * that audience is different than the authorized party. It MAY be included
125      * even when the authorized party is the same as the sole audience. The azp
126      * value is a case sensitive string containing a StringOrURI value.
127      */
128     private String azp;
129 
130     /**
131      * REQUIRED. Expiration time on or after which the ID Token MUST NOT be
132      * accepted for processing. The processing of this parameter requires that
133      * the current date/time MUST be before the expiration date/time listed in
134      * the value. Implementers MAY provide for some small leeway, usually no
135      * more than a few minutes, to account for clock skew. Its value is a JSON
136      * number representing the number of seconds from 1970-01-01T0:0:0Z as
137      * measured in UTC until the date/time. See RFC 3339 [RFC3339] for details
138      * regarding date/times in general and UTC in particular.
139      */
140     @XmlElement(required = true)
141     private long exp;
142 
143     /**
144      * REQUIRED. Time at which the JWT was issued. Its value is a JSON number
145      * representing the number of seconds from 1970-01-01T0:0:0Z as measured in
146      * UTC until the date/time.
147      */
148     @XmlElement(required = true)
149     private long iat;
150 
151     /**
152      * REQUIRED. Issuer Identifier for the Issuer of the response. The iss value
153      * is a case sensitive URL using the https scheme that contains scheme,
154      * host, and optionally, port number and path components and no query or
155      * fragment components.
156      */
157     @XmlElement(required = true)
158     private String iss;
159 
160     /**
161      * String value used to associate a Client session with an ID Token, and to
162      * mitigate replay attacks. The value is passed through unmodified from the
163      * Authentication Request to the ID Token. If present in the ID Token,
164      * Clients MUST verify that the nonce Claim Value is equal to the value of
165      * the nonce parameter sent in the Authentication Request. If present in the
166      * Authentication Request, Authorization Servers MUST include a nonce Claim
167      * in the ID Token with the Claim Value being the nonce value sent in the
168      * Authentication Request. Authorization Servers SHOULD perform no other
169      * processing on nonce values used. The nonce value is a case sensitive
170      * string.
171      */
172     private String nonce;
173 
174     /**
175      * REQUIRED. Subject Identifier. A locally unique and never reassigned
176      * identifier within the Issuer for the End-User, which is intended to be
177      * consumed by the Client, e.g., 24400320 or
178      * AItOawmwtWwcT0k51BayewNvutrJUqsvl6qs7A4. It MUST NOT exceed 255 ASCII
179      * characters in length. The sub value is a case sensitive string.
180      */
181     @XmlElement(required = true)
182     private String sub;

260     /**
261      * Checks if the exp value before or after the current time.

270     /**
271      * Sets the Issuing time and the expiration values based on the current time
272      * and expiration specified.
273      *
274      * @param expirationInSeconds
275      */
276 
277     public void resetIssueAndExpiration(final int expirationInSeconds) {
278 
279         iat = System.currentTimeMillis() / 1000;
280         exp = iat + expirationInSeconds;
281     }
