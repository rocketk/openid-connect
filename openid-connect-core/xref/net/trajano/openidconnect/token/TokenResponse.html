92      public void setScopes(Set<Scope> scopes) {
93  
94          StringBuilder b = new StringBuilder();
95          Iterator<Scope> i = scopes.iterator();
96          b.append(i.next());
97          while (i.hasNext()) {
98              b.append(' ');
99              b.append(i.next());
100         }
101         scope = b.toString();
102 
103     }
104 
105     public Set<Scope> getScopes() {
106 
107         Set<Scope> scopes = new HashSet<>();
108         for (final String scopePart : scope.split("\\s")) {
109             scopes.add(Scope.valueOf(scopePart));
110         }
111         return scopes;
112 
113     }
