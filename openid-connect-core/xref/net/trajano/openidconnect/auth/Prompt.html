1   package net.trajano.openidconnect.auth;

3   public enum Prompt {
4   
5       /**
6        * The Authorization Server SHOULD prompt the End-User for consent
7        * before returning information to the Client. If it cannot obtain
8        * consent, it MUST return an error, typically consent_required.
9        */
10      consent,
11  
12      /**
13       * The Authorization Server SHOULD prompt the End-User for
14       * reauthentication. If it cannot reauthenticate the End-User, it MUST
15       * return an error, typically login_required.
16       */
17      login,
18  
19      /**
20       * The Authorization Server MUST NOT display any authentication or
21       * consent user interface pages. An error is returned if an End-User is
22       * not already authenticated or the Client does not have pre-configured
23       * consent for the requested Claims or does not fulfill other conditions
24       * for processing the request. The error code will typically be
25       * login_required, interaction_required, or another code defined in
26       * Section 3.1.2.6. This can be used as a method to check for existing
27       * authentication and/or consent.
28       */
29      none,
30  
31      /**
32       * The Authorization Server SHOULD prompt the End-User to select a user
33       * account. This enables an End-User who has multiple accounts at the
34       * Authorization Server to select amongst the multiple accounts that
35       * they might have current sessions for. If it cannot obtain an account
36       * selection choice made by the End-User, it MUST return an error,
37       * typically account_selection_required.
38       */
39      select_account
40  
41  }
