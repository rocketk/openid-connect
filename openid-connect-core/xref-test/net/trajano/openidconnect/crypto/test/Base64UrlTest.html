3   import java.math.BigInteger;
4   
5   import net.trajano.openidconnect.crypto.Encoding;
6   
7   import org.junit.Assert;
8   import org.junit.Test;
9   
10  public class Base64UrlTest {
11  
12      /**
13       * <p>
14       * Based on <a href=
15       * "http://self-issued.info/docs/draft-ietf-jose-json-web-algorithms-35.html#eRSADef"
16       * >definiton of e</a>.
17       * </p>
18       * <blockquote> For instance, when representing the value 65537, the octet
19       * sequence to be base64url encoded MUST consist of the three octets [1, 0,
20       * 1]; the resulting representation for this value is "AQAB". </blockquote>
21       */
22      @Test
23      public void test65537() {
24  
25          Assert.assertEquals(65537, Encoding.base64urlDecodeUint("AQAB")
26                  .intValue());
27  
28          Assert.assertEquals("AQAB", Encoding.base64EncodeUint(new BigInteger("65537")));
29  
30      }
31  
32      /**
33       * Tests the zero value as per <a href=
34       * "http://self-issued.info/docs/draft-ietf-jose-json-web-algorithms-35.html#Terminology"
35       * >Base64urlUInt terminology</a>.
36       */
37      @Test
38      public void testZeroDecode() {
39  
40          Assert.assertEquals(BigInteger.ZERO, Encoding.base64urlDecodeUint("AA"));
41  
42      }
43  
44      @Test
45      public void testZeroEncode() {
46  
47          Assert.assertEquals("AA", Encoding.base64EncodeUint(BigInteger.ZERO));
48      }
49  }
