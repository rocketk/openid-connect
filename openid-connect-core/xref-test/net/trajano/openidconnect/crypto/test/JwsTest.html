29      @Test
30      public void testWithHS256() throws Exception {
31  
32          Random r = new Random();
33  
34          byte[] macKey = new byte[128];
35          final JsonWebKey jwk = new OctWebKey(macKey);
36  
37          byte[] payload = new byte[60000];
38          r.nextBytes(payload);
39  
40          JsonWebTokenBuilder b = new JsonWebTokenBuilder();
41          b.payload(payload);
42          b.alg("HS256");
43          b.jwk(jwk);
44          b.compress(true);
45          JsonWebToken jws = b.build();
46  
47          JsonWebTokenProcessor p = new JsonWebTokenProcessor(jws).allowJwkToBeSet(true)
48                  .jwk(jwk);
49          Assert.assertArrayEquals(payload, p.getPayload());
50      }
