24  /**
25   * <p>
26   * The UserInfo Endpoint is an OAuth 2.0 Protected Resource that returns Claims
27   * about the authenticated End-User. To obtain the requested Claims about the
28   * End-User, the Client makes a request to the UserInfo Endpoint using an Access
29   * Token obtained through OpenID Connect Authentication. These Claims are
30   * normally represented by a JSON object that contains a collection of name and
31   * value pairs for the Claims.
32   * </p>
33   * <p>
34   * Communication with the UserInfo Endpoint MUST utilize TLS. See Section 16.17
35   * for more information on using TLS.
36   * </p>
37   * <p>
38   * The UserInfo Endpoint MUST support the use of the HTTP GET and HTTP POST
39   * methods defined in RFC 2616 [RFC2616].
40   * </p>
41   * <p>
42   * The UserInfo Endpoint MUST accept Access Tokens as OAuth 2.0 Bearer Token
43   * Usage [RFC6750].
44   * </p>
45   * <p>
46   * The UserInfo Endpoint SHOULD support the use of Cross Origin Resource Sharing
47   * (CORS) [CORS] and or other methods as appropriate to enable Java Script
48   * Clients to access the endpoint.
49   * </p>
50   */
51  @Path("profile")
52  @Produces(MediaType.APPLICATION_JSON)
53  public class UserinfoEndpoint {
54  
55      @EJB
56      private TokenProvider tokenProvider;
57  
58      @EJB
59      private KeyProvider keyProvider;
60  
61      private UserinfoProvider userinfoProvider;
62  
63      /**
64       * <p>
65       * The Client sends the UserInfo Request using either HTTP GET or HTTP POST.
66       * The Access Token obtained from an OpenID Connect Authentication Request
67       * MUST be sent as a Bearer Token, per Section 2 of OAuth 2.0 Bearer Token
68       * Usage [RFC6750].
69       * </p>
70       * <p>
71       * It is RECOMMENDED that the request use the HTTP GET method and the Access
72       * Token be sent using the Authorization header field.
73       * </p>
74       *
75       * @param req
76       * @return
77       */
78      @GET
79      public Response getOp(@Context final HttpServletRequest req) {
80  
81          return op(req);
82      }
83  
84      @POST
85      public Response op(@Context final HttpServletRequest req) {
86  
87          final String accessToken = AuthorizationUtil.processBearer(req);
88          IdTokenResponse byAccessToken = tokenProvider.getByAccessToken(accessToken);
89          if (byAccessToken == null) {
90              return Response.status(400)
91                      .entity(new ErrorResponse(ErrorCode.access_denied, "unable to retrieve id token"))
92                      .build();
93          }
94          IdToken idToken = byAccessToken.getIdToken(keyProvider.getPrivateJwks());
95          JsonObject claims = tokenProvider.getClaimsByAccessToken(accessToken);
96  
97          Userinfo userinfo = userinfoProvider.getUserinfo(idToken);
98  
99          if (claims != null && claims.containsKey("userinfo")) {
100             filterUserInfo(userinfo, claims.getJsonObject("userinfo"));
101         }
102 
103         return Response.ok(userinfo)
104                 .build();
105     }
106 
107     private void filterUserInfo(Userinfo userinfo,
108             JsonObject userinfoClaims) {
109 
110         if (!userinfoClaims.containsKey("updated_at")) {
111             userinfo.setUpdatedAt(null);
112         }
113         if (!userinfoClaims.containsKey("email")) {
114             userinfo.setEmail(null);
115         }
116         if (!userinfoClaims.containsKey("email_verified")) {
117             userinfo.setEmailVerified(null);
118         }
119         if (!userinfoClaims.containsKey("name")) {
120             userinfo.setName(null);
121         }
122         if (!userinfoClaims.containsKey("given_name")) {
123             userinfo.setGivenName(null);
124         }
125         if (!userinfoClaims.containsKey("family_name")) {
126             userinfo.setFamilyName(null);
127         }
128         if (!userinfoClaims.containsKey("middle_name")) {
129             userinfo.setMiddleName(null);
130         }
131         if (!userinfoClaims.containsKey("nickname")) {
132             userinfo.setNickname(null);
133         }
134         if (!userinfoClaims.containsKey("website")) {
135             userinfo.setWebsite(null);
136         }
137         if (!userinfoClaims.containsKey("picture")) {
138             userinfo.setPicture(null);
139         }
140         if (!userinfoClaims.containsKey("gender")) {
141             userinfo.setGender(null);
142         }
143         if (!userinfoClaims.containsKey("address")) {
144             userinfo.setAddress(null);
145         }
146         if (!userinfoClaims.containsKey("profile")) {
147             userinfo.setProfile(null);
148         }
149         if (!userinfoClaims.containsKey("birthdate")) {
150             userinfo.setBirthdate(null);
151         }
152         if (!userinfoClaims.containsKey("zoneinfo")) {
153             userinfo.setZoneinfo(null);
154         }
155         if (!userinfoClaims.containsKey("locale")) {
156             userinfo.setLocale(null);
157         }
158         if (!userinfoClaims.containsKey("preferred_username")) {
159             userinfo.setPreferredUsername(null);
160         }
161         if (!userinfoClaims.containsKey("phone_number")) {
162             userinfo.setPhoneNumber(null);
163         }
164         if (!userinfoClaims.containsKey("phone_number_verified")) {
165             userinfo.setPhoneNumberVerified(null);
166         }
167     }
168

170     public void setUserinfoProvider(final UserinfoProvider userinfoProvider) {
171 
172         this.userinfoProvider = userinfoProvider;
173     }
