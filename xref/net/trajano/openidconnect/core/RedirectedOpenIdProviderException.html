16  /**
17   * Renders the {@link OpenIdConnectException} as a redirect with an optional
18   * state value. It will only render {@value } OpenIdConnectKey#ERROR},
19   * {@value OpenIdConnectKey#ERROR_DESCRIPTION} and {@value }
20   * {@value OpenIdConnectKey#ERROR_URI} from the response.
21   * 
22   * @author Archimedes Trajano
23   */
24  public class RedirectedOpenIdProviderException extends WebApplicationException {
25  
26      /**
27       *
28       */
29      private static final long serialVersionUID = -8684305650194342408L;
30  
31      private static Response responseBuilder(final URI redirectUri,
32              final ErrorResponse errorResponse,
33              final String state) {
34  
35          final UriBuilder b = UriBuilder.fromUri(redirectUri)
36                  .queryParam(ERROR, errorResponse.getError());
37          if (errorResponse.getErrorDescription() != null) {
38              b.queryParam(ERROR_DESCRIPTION, errorResponse.getErrorDescription());
39          }
40          if (errorResponse.getErrorUri() != null) {
41              b.queryParam(ERROR_URI, errorResponse.getErrorUri());
42          }
43          if (state != null) {
44              b.queryParam(STATE, state);
45          }
46          return Response.temporaryRedirect(b.build())
47                  .build();
48      }
49  
50      /**
51       * Convenience constructor as this exception type is generally part of
52       * authentication.
53       * 
54       * @param authenticationRequest
55       * @param errorResponse
56       */
57      public RedirectedOpenIdProviderException(final AuthenticationRequest authenticationRequest, ErrorResponse errorResponse) {
58  
59          this(authenticationRequest.getRedirectUri(), errorResponse, authenticationRequest.getState());
60      }
61  
62      public RedirectedOpenIdProviderException(URI redirectUri, ErrorResponse errorResponse) {
63  
64          this(redirectUri, errorResponse, null);
65      }
66  
67      public RedirectedOpenIdProviderException(URI redirectUri, ErrorResponse errorResponse, String state) {
68  
69          super(responseBuilder(redirectUri, errorResponse, state));
70      }
71  }
