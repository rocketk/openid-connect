17  /**
18   * <p>
19   * The response is a set of Claims about the OpenID Provider's configuration,
20   * including all necessary endpoints and public key location information. A
21   * successful response MUST use the 200 OK HTTP status code and return a JSON
22   * object using the application/json content type that contains a set of Claims
23   * as its members that are a subset of the Metadata values defined in Section 3.
24   * Other Claims MAY also be returned.
25   * </p>
26   * <p>
27   * Claims that return multiple values are represented as JSON arrays. Claims
28   * with zero elements MUST be omitted from the response.
29   * </p>
30   * <p>
31   * An error response uses the applicable HTTP status code value.
32   * </p>
33   */
34  @XmlRootElement
35  public class OpenIdProviderConfiguration {

144     /**
145      * URL of the OP's JSON Web Key Set [JWK] document. This contains the
146      * signing key(s) the RP uses to validate signatures from the OP. The JWK
147      * Set MAY also contain the Server's encryption key(s), which are used by
148      * RPs to encrypt requests to the Server. When both signing and encryption
149      * keys are made available, a use (Key Use) parameter value is REQUIRED for
150      * all keys in the referenced JWK Set to indicate each key's intended usage.
151      * Although some algorithms allow the same key to be used for both
152      * signatures and encryption, doing so is NOT RECOMMENDED, as it is less
153      * secure. The JWK x5c parameter MAY be used to provide X.509
154      * representations of keys provided. When used, the bare key values MUST
155      * still be present and MUST match those in the certificate.
156      */
157     @XmlElement(name = "jwks_uri")
158     private URI jwksUri;
