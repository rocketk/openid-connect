42      /**
43       * Builds the curve using the values from NIST.
44       *
45       * @param pS
46       *            the p value as a string. This is the prime modulus.
47       * @param rS
48       *            the r value as a string. This is the prime order.
49       * @param sS
50       *            the s value as a string. This is a base-16 value. The 160-bit
51       *            input seed s to the SHA-1 based algorithm
52       * @param cS
53       *            output of the SHA-1 based algorithm
54       * @param bS
55       *            coefficient
56       * @param gxS
57       *            The base point x coordinate G<sub>x</sub>
58       * @param gyS
59       *            The base point y coordinate G<sub>y</sub>
60       */
61      private NamedEllipticCurve(final String pS, final String rS, final String sS, final String cS, final String bS, final String gxS, final String gyS) {
62  
63          final BigInteger p = new BigInteger(pS);
64          final BigInteger r = new BigInteger(rS);
65          final BigInteger s = new BigInteger(sS, 16);
66  
67          final EllipticCurve curve = new EllipticCurve(new ECFieldFp(p), r, s);
68  
69          final BigInteger gx = new BigInteger(gxS, 16);
70          final BigInteger gy = new BigInteger(gyS, 16);
71          final ECPoint g = new ECPoint(gx, gy);
72  
73          final BigInteger b = new BigInteger(bS, 16);
74          final BigInteger c = new BigInteger(cS, 16);
75          assert validate(b, c, p);
76          parameterSpec = new ECParameterSpec(curve, g, b, 1);
77  
78      }
79  
80      private static boolean validate(BigInteger b,
81              BigInteger c,
82              BigInteger p) {
83  
84          return ((b.pow(2)).multiply(c)
85                  .add(new BigInteger("27")).mod(p)).equals(BigInteger.ZERO);
86      }
87  
88      public ECParameterSpec toECParameterSpec() {

90          return parameterSpec;
