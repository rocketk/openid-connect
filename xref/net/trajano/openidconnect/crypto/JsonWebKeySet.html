3   import java.security.GeneralSecurityException;
4   import java.security.Key;
5   import java.util.HashMap;
6   import java.util.Map;
7   
8   import javax.xml.bind.annotation.XmlAccessType;
9   import javax.xml.bind.annotation.XmlAccessorType;
10  import javax.xml.bind.annotation.XmlElement;
11  import javax.xml.bind.annotation.XmlRootElement;
12  
13  @XmlRootElement
14  @XmlAccessorType(XmlAccessType.NONE)
15  public class JsonWebKeySet {
16  
17      private final Map<String, JsonWebKey> keys = new HashMap<>();
18      private final Map<String, JsonWebKey> signingKeys = new HashMap<>();
19  
20      public void add(final JsonWebKey jwk) {
21  
22          keys.put(jwk.getKid(), jwk);
23          if (jwk.getKty() == KeyType.RSA || jwk.getKty() == KeyType.EC) {
24              signingKeys.put(jwk.getKid(), jwk);
25          }
26      }
27  
28      @XmlElement(name = "keys", required = true)
29      public JsonWebKey[] getKeys() {
30  
31          return keys.values()
32                  .toArray(new JsonWebKey[0]);
33      }
34      
35  
36      public JsonWebKey[] getSigningKeys() {
37  
38          return signingKeys.values()
39                  .toArray(new JsonWebKey[0]);
40      }
41  
42  
43      public Key getKey(String kid) throws GeneralSecurityException {
44  
45          return getJwk(kid).toJcaKey();
46      }
47  
48      public JsonWebKey getJwk(String kid) {
49  
50          return keys.get(kid);
51  
52      }
53  }
