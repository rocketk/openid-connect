13  public class JsonWebTokenProcessor {
14  
15      private String alg = JsonWebToken.ALG_NONE;
16  
17      /**
18       * Flag to indicate whether jwk can be set directly.
19       */
20      private boolean allowJwkToBeSet = false;
21  
22      private final JsonWebTokenCrypto crypto = JcaJsonWebTokenCrypto.getInstance();
23  
24      private String enc = null;
25  
26      private JsonWebToken jsonWebToken;
27  
28      private JsonWebKey jwk = null;
29  
30      private String kid = null;
31  
32      /**
33       * Flag to indicate that signature check is required.
34       */
35      private boolean signatureCheck = true;
36  
37      public JsonWebTokenProcessor(final JsonWebToken jsonWebToken) {
38  
39          this.jsonWebToken = jsonWebToken;
40          alg = jsonWebToken.getAlg();
41          enc = jsonWebToken.getEnc();
42          kid = jsonWebToken.getKid();
43  
44      }

46      public JsonWebTokenProcessor(final String serialization) throws IOException {
47  
48          this(new JsonWebToken(serialization));
49      }

51      public JsonWebTokenProcessor allowJwkToBeSet(final boolean flag) throws IOException {
52  
53          allowJwkToBeSet = flag;
54          return this;
55      }
56  
57      /**
58       * Gets the payload JSON object.
59       *
60       * @return payload a a JSON object.
61       * @throws GeneralSecurityException
62       */
63      public JsonObject getJsonPayload() throws GeneralSecurityException {
64  
65          final JsonReader r = Json.createReader(new ByteArrayInputStream(getPayload()));
66          return r.readObject();
67  
68      }
69  
70      public byte[] getPayload() throws GeneralSecurityException {
71  
72          byte[] payload;
73          if (JsonWebToken.ALG_NONE.equals(alg)) {
74              payload = jsonWebToken.getPayload(0);
75          } else if (enc != null) {
76              if (jsonWebToken.getNumberOfPayloads() != 4) {
77                  throw new GeneralSecurityException("invalid number of payloads in JWT for JWE");
78              }
79              payload = crypto.getJWEPayload(jsonWebToken, jwk);
80          } else if (enc == null && alg != null && signatureCheck) {
81              if (jsonWebToken.getNumberOfPayloads() != 2) {
82                  throw new GeneralSecurityException("invalid number of payloads in JWT for JWS");
83              }
84              if (jwk == null) {
85                  throw new GeneralSecurityException("No JWK available to validate signature");
86              }
87              payload = crypto.getJWSPayload(jsonWebToken, jwk, alg);
88          } else if (enc == null && alg != null && !signatureCheck) {
89              if (jsonWebToken.getNumberOfPayloads() != 2) {
90                  throw new GeneralSecurityException("invalid number of payloads in JWT for JWS");
91              }
92              payload = jsonWebToken.getPayload(0);
93          } else {
94              throw new GeneralSecurityException("invalid JOSE header");
95          }
96          if ("DEF".equals(jsonWebToken.getZip())) {
97              try {
98                  return crypto.inflate(payload);
99              } catch (IOException e) {
100                 throw new GeneralSecurityException(e);
101             }
102         } else {
103             return payload;
104         }
105 
106     }
107 
108     public JsonWebTokenProcessor jwk(final JsonWebKey jwk) throws IOException {
109 
110         if (!allowJwkToBeSet) {
111             throw new IOException("jwk cannot be explicitly set");
112         }
113         this.jwk = jwk;
114         return this;
115 
116     }
117 
118     public JsonWebTokenProcessor jwks(final JsonWebKeySet jwks) throws IOException {
119 
120         if (kid != null) {
121             jwk = jwks.getJwk(kid);
122         } else if (jwks.getKeys().length == 1) {
123             jwk = jwks.getKeys()[0];
124         }
125         return this;
126 
127     }

129     /**
130      * <p>
131      * Checks if the JWK is already set for the processor. The value is not set
132      * if the {@link #jwk(JsonWebKey)} is not called or the
133      * {@link #jwks(JsonWebKeySet)} is not called or does not contain the key
134      * specified by the {@link #kid}.
135      * </p>
136      * <p>
137      * If the algorithm is none then it will also return <code>true</code> as a
138      * JWK is not needed.
139      * </p>
140      * 
141      * @return jwk is set.
142      */
143     public boolean isJwkAvailable() {
144 
145         return jwk != null || JsonWebToken.ALG_NONE.equals(alg);
146     }
147 
148     /**
149      * Enable or disable signature checks. Signature checks are enabled by
150      * default. Signature checks can be disabled for scenarios where the
151      * signature had already been validated previously in the process and the
152      * JWT was stored as is rather than having the extracted payload kept.
153      */
154     public JsonWebTokenProcessor signatureCheck(boolean b) {
155 
156         signatureCheck = b;
157         return this;
158 
159     }
