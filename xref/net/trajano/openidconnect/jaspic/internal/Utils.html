11  /**
12   * Utility methods. Normally these would be in a separate JAR file like
13   * commons-lang, but to prevent complications during installation such as
14   * requiring to install additional JAR files, this class was created.
15   *
16   * @author Archimedes Trajano
17   */
18  public final class Utils {
19  
20      /**
21       * Checks if the request uses the GET method.
22       *
23       * @param req
24       *            request
25       * @return <code>true</code> if the request uses the GET method.
26       * @deprecated use {@link ValidateContext#isGetRequest()}
27       */
28      @Deprecated
29      public static boolean isGetRequest(final HttpServletRequest req) {
30  
31          return "GET".equals(req.getMethod());
32      }
33  
34      /**
35       * Checks if the request uses the HEAD method.
36       *
37       * @param req
38       *            request
39       * @return <code>true</code> if the request uses the HEAD method.
40       */
41      public static boolean isHeadRequest(final HttpServletRequest req) {
42  
43          return "HEAD".equals(req.getMethod());
44      }
45  
46      /**
47       * Checks if string is null or empty.
48       *
49       * @param s
50       *            string to test
51       * @return true if string is null or empty.
52       */
53      public static boolean isNullOrEmpty(final String s) {
54  
55          return s == null || s.trim()
56              .length() == 0;
57      }
58  
59      /**
60       * Checks if the request is to retrieve data (i.e. "GET" or "HEAD" method).
61       *
62       * @param req
63       *            request
64       * @return <code>true</code> if the request uses the GET or HEAD method.
65       */
66      public static boolean isRetrievalRequest(final HttpServletRequest req) {
67  
68          return isGetRequest(req) || isHeadRequest(req);
69      }
70  
71      /**
72       * Validates the ID Token.
73       *
74       * @param clientId
75       *            client ID
76       * @param idTokenJson
77       *            ID Token JSON.
78       * @throws GeneralSecurityException
79       */
80      public static void validateIdToken(final String clientId,
81          final JsonObject idTokenJson,
82          final String nonce,
83          final String accessToken) throws GeneralSecurityException {
84  
85          // TODO handle multiple audiences
86          if (!clientId.equals(idTokenJson.getString("aud"))) {
87              throw new GeneralSecurityException(String.format("invalid 'aud' got' %s' expected '%s'", idTokenJson.getString("aud"), clientId));
88          }
89          if (nonce != null && !nonce.equals(idTokenJson.getString("nonce"))) {
90              throw new GeneralSecurityException(String.format("invalid 'nonce' got' %s' expected '%s'", idTokenJson.getString("nonce"), clientId));
91          }
92          if (idTokenJson.containsKey("azp") && !clientId.equals(idTokenJson.getString("azp"))) {
93              throw new GeneralSecurityException(String.format("invalid 'azp' got' %s' expected '%s'", idTokenJson.getString("azp"), clientId));
94          }
95          if (idTokenJson.containsKey("exp")) {
96              final long delta = System.currentTimeMillis() - idTokenJson.getInt("exp") * 1000L;
97              if (delta >= 0) {
98                  throw new GeneralSecurityException("expired " + delta + "ms ago");
99              }
100         }
101         if (idTokenJson.containsKey("at_hash")) {
102             final MessageDigest digest = MessageDigest.getInstance("SHA-256");
103             final byte[] digestedBytes = digest.digest(accessToken.getBytes());
104             if (!Encoding.base64urlEncode(digestedBytes, 0, 128 / 8)
105                 .equals(idTokenJson.getString("at_hash"))) {
106                 throw new GeneralSecurityException("access token hash mismatch");
107             }
108         }
109     }
