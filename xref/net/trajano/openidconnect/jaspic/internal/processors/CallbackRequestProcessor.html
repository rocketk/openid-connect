60      /**
61       * Checks to see whether redirection end point callback the
62       * {@link ServerAuthModule} is called by the user agent. This is indicated
63       * by the presence of a <code>code</code> and a <code>state</code> on the
64       * URL. The user agent would be a web browser that got a redirect or
65       * automatic form post sent by the OP.
66       */
67      @Override
68      public boolean canValidateRequest(final ValidateContext context) {
69  
70          return context.isSecure() && context.isRequestUri(OpenIdConnectAuthModule.REDIRECTION_ENDPOINT_URI_KEY) && !isNullOrEmpty(context.getReq()
71              .getParameter(CODE)) && !isNullOrEmpty(context.getReq()
72                  .getParameter(STATE));
73  
74      }

76      /**
77       * Sends a request to the token endpoint to get the token for the code.
78       *
79       * @param req
80       *            servlet request
81       * @param oidProviderConfig
82       *            OpenID provider config
83       * @return token response
84       */
85      private IdTokenResponse getToken(final String grantKey,
86          final String grant,
87          final GrantType grantType,
88          final ValidateContext context) throws IOException {
89  
90          final MultivaluedMap<String, String> requestData = new MultivaluedHashMap<>();
91          requestData.putSingle(grantKey, grant);
92          requestData.putSingle(GRANT_TYPE, grantType.name());
93          requestData.putSingle(REDIRECT_URI, context.getUri(OpenIdConnectAuthModule.REDIRECTION_ENDPOINT_URI_KEY)
94              .toASCIIString());
95  
96          try {
97              final String authorization = "Basic " + Encoding.base64Encode(context.getOption(OpenIdConnectKey.CLIENT_ID) + ":" + context.getOption(OpenIdConnectKey.CLIENT_SECRET));
98              final IdTokenResponse authorizationTokenResponse = context.target(context.getOpenIDProviderConfig()
99                  .getTokenEndpoint())
100                 .request(MediaType.APPLICATION_JSON_TYPE)
101                 .header("Authorization", authorization)
102                 .post(Entity.form(requestData), IdTokenResponse.class);
103             if (LOG.isLoggable(Level.FINEST)) {
104                 LOG.finest("authorization token response =  " + authorizationTokenResponse);
105             }
106             return authorizationTokenResponse;
107         } catch (final BadRequestException e) {
108             // workaround for google that does not support BASIC authentication
109             // on their endpoint.
110             requestData.putSingle(CLIENT_ID, context.getOption(OpenIdConnectKey.CLIENT_ID));
111             requestData.putSingle(CLIENT_SECRET, context.getOption(OpenIdConnectKey.CLIENT_SECRET));
112             final IdTokenResponse authorizationTokenResponse = context.target(context.getOpenIDProviderConfig()
113                 .getTokenEndpoint())
114                 .request(MediaType.APPLICATION_JSON_TYPE)
115                 .post(Entity.form(requestData), IdTokenResponse.class);
116             if (LOG.isLoggable(Level.FINEST)) {
117                 LOG.finest("authorization token response =  " + authorizationTokenResponse);
118             }
119             return authorizationTokenResponse;
120         }
121     }

123     /**
124      * Gets the web keys from the options and the OpenID provider configuration.
125      * This may be overridden by clients.
126      *
127      * @param config
128      *            provider configuration
129      * @return web keys
130      * @throws GeneralSecurityException
131      *             wraps exceptions thrown during processing
132      */
133     private JsonWebKeySet getWebKeys(final ValidateContext context) throws GeneralSecurityException {
134 
135         return context.target(context.getOpenIDProviderConfig()
136             .getJwksUri())
137             .request(MediaType.APPLICATION_JSON_TYPE)
138             .get(JsonWebKeySet.class);
139     }
140 
141     /**
142      * Workaround for the issuer value for Google. This was documented in
143      * 15.6.2. of the spec. In which case if the issuer does not start with
144      * https:// it will prepend it.
145      *
146      * @param issuer
147      *            issuer
148      * @return updated issuer
149      */
150     private String googleWorkaround(final String issuer) {
151 
152         if (issuer.startsWith(HTTPS_PREFIX)) {
153             return issuer;
154         }
155         return HTTPS_PREFIX + issuer;
156     }
157 
158     /**
159      * Updates the principal for the subject. This is done through the
160      * callbacks.
161      *
162      * @param subject
163      *            subject
164      * @param jwtPayload
165      *            JWT payload
166      * @throws AuthException
167      * @throws GeneralSecurityException
168      */
169     private void updateSubjectPrincipal(final Subject subject,
170         final JsonObject jwtPayload,
171         final ValidateContext context) throws GeneralSecurityException {
172 
173         try {
174             final String iss = googleWorkaround(jwtPayload.getString("iss"));
175             context.getHandler()
176                 .handle(new Callback[] {
177                     new CallerPrincipalCallback(subject, UriBuilder.fromUri(iss)
178                         .userInfo(jwtPayload.getString("sub"))
179                         .build()
180                         .toASCIIString()),
181                     new GroupPrincipalCallback(subject, new String[] {
182                         iss
183                             })
184             });
185         } catch (final IOException
186             | UnsupportedCallbackException e) {
187             // Should not happen
188             LOG.log(Level.SEVERE, "updatePrincipalException", e.getMessage());
189             LOG.throwing(this.getClass()
190                 .getName(), "updateSubjectPrincipal", e);
191             throw new AuthException(MessageFormat.format(Log.r("updatePrincipalException"), e.getMessage()));
192         }
193     }

195     @Override
196     public AuthStatus validateRequest(final ValidateContext context) throws IOException,
197         GeneralSecurityException {
198 
199         final OpenIdProviderConfiguration oidProviderConfig = context.getOpenIDProviderConfig();
200         final IdTokenResponse token = getToken(OpenIdConnectKey.CODE, context.getReq()
201             .getParameter(OpenIdConnectKey.CODE), GrantType.authorization_code, context);
202         final net.trajano.openidconnect.crypto.JsonWebKeySet webKeys = getWebKeys(context);
203 
204         LOG.log(Level.FINEST, "tokenValue", token);
205         final JsonObject claimsSet = new JsonWebTokenProcessor(token.getEncodedIdToken()).jwks(webKeys)
206             .getJsonPayload();
207 
208         final String nonceCookie = context.getCookie(OpenIdConnectAuthModule.NET_TRAJANO_AUTH_NONCE);
209         final String nonce;
210         if (nonceCookie != null) {
211             nonce = new String(CipherUtil.decrypt(Encoding.base64urlDecode(nonceCookie), context.getSecret()), "US-ASCII");
212         } else {
213             nonce = null;
214         }
215 
216         validateIdToken(context.getOption(CLIENT_ID), claimsSet, nonce, token.getAccessToken());
217 
218         context.deleteCookie(OpenIdConnectAuthModule.NET_TRAJANO_AUTH_NONCE);
219 
220         final String iss = googleWorkaround(claimsSet.getString("iss"));
221         final String issuer = googleWorkaround(oidProviderConfig.getIssuer());
222         if (!iss.equals(issuer)) {
223             LOG.log(Level.SEVERE, "issuerMismatch", new Object[] {
224                 iss,
225                 issuer
226             });
227             throw new GeneralSecurityException(MessageFormat.format(Log.r("issuerMismatch"), iss, issuer));
228         }
229         updateSubjectPrincipal(context.getClientSubject(), claimsSet, context);
230 
231         final TokenCookie tokenCookie;
232         if (oidProviderConfig.getUserinfoEndpoint() != null && Pattern.compile("\\bprofile\\b")
233             .matcher(context.getOption(OpenIdConnectKey.SCOPE))
234             .find()) {
235             final Response userInfoResponse = context.target(oidProviderConfig.getUserinfoEndpoint())
236                 .request(MediaType.APPLICATION_JSON_TYPE)
237                 .header("Authorization", token.getTokenType() + " " + token.getAccessToken())
238                 .get();
239             if (userInfoResponse.getStatus() == 200) {
240                 tokenCookie = new TokenCookie(token.getAccessToken(), token.getRefreshToken(), claimsSet, token.getEncodedIdToken(), userInfoResponse.readEntity(JsonObject.class));
241             } else {
242                 LOG.log(Level.WARNING, "unableToGetProfile");
243                 tokenCookie = new TokenCookie(claimsSet, token.getEncodedIdToken());
244             }
245         } else {
246             tokenCookie = new TokenCookie(claimsSet, token.getEncodedIdToken());
247         }
248 
249         context.saveIdTokenCookie(tokenCookie);
250         context.saveAgeCookie();
251         context.redirectToState();
252         return AuthStatus.SEND_SUCCESS;
253     }
