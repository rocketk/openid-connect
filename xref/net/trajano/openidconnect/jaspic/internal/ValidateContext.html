54      public ValidateContext(final Client client, final Subject clientSubject, final boolean mandatory, final Map<String, String> options, final HttpServletRequest req, final HttpServletResponse resp, final TokenCookie tokenCookie, final String cookieContext, final CallbackHandler handler) {
55  
56          this.client = client;
57          this.clientSubject = clientSubject;
58          this.mandatory = mandatory;
59          this.options = options;
60          this.req = req;
61          this.resp = resp;
62          this.tokenCookie = tokenCookie;
63          this.handler = handler;
64  
65          if (isNullOrEmpty(cookieContext)) {
66              this.cookieContext = req.getContextPath();
67          } else {
68              this.cookieContext = cookieContext;
69          }
70  
71      }

74       * Deletes the authentication cookies.
75       */
76      public void deleteAuthCookies() {
77  
78          for (final String cookieName : AUTH_COOKIE_NAMES) {
79              final Cookie deleteCookie = new Cookie(cookieName, "");
80              deleteCookie.setMaxAge(0);
81              deleteCookie.setSecure(true);
82              deleteCookie.setPath(cookieContext);
83              resp.addCookie(deleteCookie);
84          }
85  
86      }

107     public String getCookie(final String cookieName) {
108 
109         final Cookie[] cookies = req.getCookies();
110         if (cookies == null) {
111             return null;
112         }
113 
114         for (final Cookie cookie : cookies) {
115             if (cookieName.equals(cookie.getName())) {
116                 return cookie.getValue();
117             }
118         }
119         return null;
120     }

141     public synchronized OpenIdProviderConfiguration getOpenIDProviderConfig() {
142 
143         if (oidConfig == null) {
144             final String issuerUri = options.get(OpenIdConnectAuthModule.ISSUER_URI_KEY);
145             if (issuerUri == null) {
146                 // LOG.log(Level.SEVERE, "missingOption", ISSUER_URI_KEY);
147                 // throw new
148                 // AuthException(MessageFormat.format(R.getString("missingOption"),
149                 // ISSUER_URI_KEY));
150             }
151             oidConfig = client.target(URI.create(issuerUri)
152                     .resolve("/.well-known/openid-configuration"))
153                     .request(MediaType.APPLICATION_JSON_TYPE)
154                     .get(OpenIdProviderConfiguration.class);
155 
156         }
157         return oidConfig;
158 
159     }

248     /**
249      * Redirects to the last state. However, if state is equivalent to the
250      * logout URI it will redirect to the root.
251      *
252      * @throws IOException
253      */
254     public void redirectToState() throws IOException {
255 
256         final String stateEncoded = req.getParameter(OpenIdConnectKey.STATE);
257         final String contextRedirectUri = Encoding.base64urlDecodeToString(stateEncoded);
258         final String targetUri = req.getContextPath() + contextRedirectUri;
259         if (targetUri.equals(options.get(OpenIdConnectAuthModule.LOGOUT_URI_KEY))) {
260             Log.fine("state was the Logout URI, redirecting to context root");
261             resp.sendRedirect(resp.encodeRedirectURL(req.getContextPath()));
262         } else {
263             resp.sendRedirect(resp.encodeRedirectURL(targetUri));
264         }
265 
266     }

268     public void saveAgeCookie() throws GeneralSecurityException,
269             IOException {
270 
271         final Cookie ageCookie = new Cookie(OpenIdConnectAuthModule.NET_TRAJANO_AUTH_AGE, Encoding.base64urlEncode(CipherUtil.encrypt(req.getRemoteAddr()
272                 .getBytes(CharSets.US_ASCII), secret)));
273         if (isNullOrEmpty(req.getParameter("expires_in"))) {
274             ageCookie.setMaxAge(3600);
275 
276         } else {
277             ageCookie.setMaxAge(Integer.parseInt(req.getParameter("expires_in")));
278         }
279         ageCookie.setPath(cookieContext);
280         ageCookie.setSecure(true);
281         ageCookie.setHttpOnly(true);
282         resp.addCookie(ageCookie);
283 
284     }

292     public void saveIdTokenCookie(final TokenCookie tokenCookie) throws GeneralSecurityException {
293 
294         final Cookie idTokenCookie = new Cookie(OpenIdConnectAuthModule.NET_TRAJANO_AUTH_ID, tokenCookie.toCookieValue(getSecret()));
295         idTokenCookie.setMaxAge(-1);
296         idTokenCookie.setSecure(true);
297         idTokenCookie.setHttpOnly(true);
298         idTokenCookie.setPath(cookieContext);
299         resp.addCookie(idTokenCookie);
300     }
