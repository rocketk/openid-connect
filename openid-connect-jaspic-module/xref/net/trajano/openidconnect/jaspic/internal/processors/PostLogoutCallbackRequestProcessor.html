3   import static net.trajano.openidconnect.core.OpenIdConnectKey.STATE;
4   import static net.trajano.openidconnect.jaspic.internal.Utils.isNullOrEmpty;
5   
6   import java.io.IOException;
7   import java.security.GeneralSecurityException;
8   
9   import javax.security.auth.message.AuthStatus;
10  import javax.security.auth.message.module.ServerAuthModule;
11  
12  import net.trajano.openidconnect.jaspic.internal.ValidateContext;
13  import net.trajano.openidconnect.jaspic.internal.ValidateRequestProcessor;
14  
15  public class PostLogoutCallbackRequestProcessor implements ValidateRequestProcessor {
16  
17      /**
18       * Checks to see whether post logout redirection end point callback the
19       * {@link ServerAuthModule} is called by the user agent. This is indicated
20       * by the presence of a <code>state</code> on the URL. The user agent would
21       * be a web browser that got a redirect or automatic form post sent by the
22       * OP.
23       *
24       * @param req
25       *            HTTP servlet request
26       * @return the module is called by the resource owner.
27       */
28      @Override
29      public boolean canValidateRequest(final ValidateContext context) {
30  
31          return context.isSecure() && context.isRequestUri("logout_redirection_endpoint") && !isNullOrEmpty(context.getReq()
32                  .getParameter(STATE));
33      }
34  
35      @Override
36      public AuthStatus validateRequest(final ValidateContext context) throws IOException,
37              GeneralSecurityException {
38  
39          context.redirectToState();
40          return AuthStatus.SEND_SUCCESS;
41      }
42  
43  }
