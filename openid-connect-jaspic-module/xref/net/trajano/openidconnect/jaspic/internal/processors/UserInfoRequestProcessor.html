3   import java.io.IOException;
4   import java.security.GeneralSecurityException;
5   
6   import javax.security.auth.message.AuthStatus;
7   import javax.ws.rs.core.MediaType;
8   
9   import net.trajano.openidconnect.jaspic.OpenIdConnectAuthModule;
10  import net.trajano.openidconnect.jaspic.internal.ValidateContext;
11  import net.trajano.openidconnect.jaspic.internal.ValidateRequestProcessor;
12  
13  public class UserInfoRequestProcessor implements ValidateRequestProcessor {
14  
15      @Override
16      public boolean canValidateRequest(final ValidateContext context) {
17  
18          if (!context.hasTokenCookie()) {
19              return false;
20          }
21  
22          return context.isSecure() && context.isGetRequest() && context.isRequestUri(OpenIdConnectAuthModule.USERINFO_URI_KEY);
23      }
24  
25      @Override
26      public AuthStatus validateRequest(final ValidateContext context) throws IOException,
27              GeneralSecurityException {
28  
29          context.setContentType(MediaType.APPLICATION_JSON);
30          context.getResp()
31                  .getWriter()
32                  .print(context.getTokenCookie()
33                          .getUserInfo());
34          return AuthStatus.SEND_SUCCESS;
35      }
36  
37  }
