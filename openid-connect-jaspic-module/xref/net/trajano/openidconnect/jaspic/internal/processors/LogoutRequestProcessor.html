1   package net.trajano.openidconnect.jaspic.internal.processors;
2   
3   import java.io.IOException;
4   import java.net.URI;
5   import java.security.GeneralSecurityException;
6   
7   import javax.security.auth.message.AuthStatus;
8   import javax.ws.rs.core.UriBuilder;
9   
10  import net.trajano.openidconnect.core.OpenIdProviderConfiguration;
11  import net.trajano.openidconnect.crypto.Encoding;
12  import net.trajano.openidconnect.jaspic.OpenIdConnectAuthModule;
13  import net.trajano.openidconnect.jaspic.internal.Log;
14  import net.trajano.openidconnect.jaspic.internal.ValidateContext;
15  import net.trajano.openidconnect.jaspic.internal.ValidateRequestProcessor;
16  
17  public class LogoutRequestProcessor implements ValidateRequestProcessor {
18  
19      /**
20       * Encoded value for the context root state.
21       */
22      private static final String CONTEXT_ROOT_STATE = Encoding.base64urlEncode("/");
23  
24      /**
25       * This only supports the scenario when logout goto uri is not available.
26       */
27      @Override
28      public boolean canValidateRequest(final ValidateContext context) {
29  
30          if (context.hasOption(OpenIdConnectAuthModule.LOGOUT_GOTO_URI_KEY)) {
31              return false;
32          }
33  
34          return context.isSecure() && context.isGetRequest() && context.isRequestUri(OpenIdConnectAuthModule.LOGOUT_URI_KEY);
35      }
36  
37      @Override
38      public AuthStatus validateRequest(final ValidateContext context) throws IOException,
39              GeneralSecurityException {
40  
41          String idTokenHint = context.getTokenCookie()
42                  .getIdTokenJwt();
43          context.deleteAuthCookies();
44          final OpenIdProviderConfiguration oidProviderConfig = context.getOpenIDProviderConfig();
45  
46          String contextPath = UriBuilder.fromUri(context.getReq()
47                  .getRequestURL()
48                  .toString())
49                  .replacePath(context.getReq()
50                          .getContextPath())
51                  .build()
52                  .toASCIIString();
53          final String referrer = context.getReq()
54                  .getHeader("Referer");
55          final String state;
56          if (referrer.startsWith(contextPath)) {
57  
58              final StringBuilder stateBuilder = new StringBuilder(referrer.substring(contextPath.length()));
59              if (context.getReq()
60                      .getQueryString() != null) {
61                  stateBuilder.append('?');
62                  stateBuilder.append(context.getReq()
63                          .getQueryString());
64              }
65              state = Encoding.base64urlEncode(stateBuilder.toString());
66          } else {
67              Log.fine("Referrer " + referrer + "does not start with context path " + contextPath + " using root context");
68              state = CONTEXT_ROOT_STATE;
69          }
70  
71          final URI redirectUri = context.getUri("logout_redirection_endpoint");
72  
73          if (oidProviderConfig.getEndSessionEndpoint() != null) {
74              UriBuilder b = UriBuilder.fromUri(oidProviderConfig.getEndSessionEndpoint())
75                      .queryParam("post_logout_redirect_uri", redirectUri)
76                      .queryParam("id_token_hint", idTokenHint)
77                      .queryParam("state", state);
78              context.getResp()
79                      .sendRedirect(b.build()
80                              .toASCIIString());
81          } else {
82              context.getResp()
83                      .sendRedirect(context.getReq()
84                              .getServletContext() + "/");
85          }
86          return AuthStatus.SEND_SUCCESS;
87      }
88  
89  }
