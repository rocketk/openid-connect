12  import net.trajano.openidconnect.crypto.Encoding;

15  /**
16   * Manages the token cookie.
17   *
18   * @author Archimedes
19   */
20  public class TokenCookie {
21  
22      /**
23       * Access token key in the tokens structure.
24       */
25      private static final String ACCESS_TOKEN_KEY = "a";
26  
27      /**
28       * Refresh token key in the tokens structure.
29       */
30      private static final String REFRESH_TOKEN_KEY = "r";
31  
32      /**
33       * Access Token.
34       */
35      private final String accessToken;
36  
37      /**
38       * ID Token.
39       */
40      private final JsonObject idToken;
41  
42      /**
43       * Original JWT for the ID token that was sent over.
44       */
45      private final String idTokenJwt;
46  
47      /**
48       * Refresh Token.
49       */
50      private final String refreshToken;
51  
52      /**
53       * User info.
54       */
55      private final JsonObject userInfo;
56  
57      /**
58       * Constructs with just the ID token.
59       *
60       * @param idToken
61       *            ID token
62       */
63      public TokenCookie(final JsonObject idToken, String idTokenJwt) {
64  
65          this("", "", idToken, idTokenJwt, null);
66      }
67  
68      /**
69       * Constructs with the cookie value.
70       *
71       * @param cookieValue
72       *            cookie value
73       * @param secret
74       *            secret key
75       * @throws GeneralSecurityException
76       */
77      public TokenCookie(final String cookieValue, final SecretKey secret) throws GeneralSecurityException {
78  
79          final String[] cookieValues = cookieValue.split("\\.");
80  
81          try {
82              final JsonObject tokens = Json.createReader(CipherUtil.buildDecryptStream(new ByteArrayInputStream(Encoding.base64urlDecode(cookieValues[0])), secret))
83                      .readObject();
84              accessToken = ((JsonString) tokens.get(ACCESS_TOKEN_KEY)).getString();
85              refreshToken = ((JsonString) tokens.get(REFRESH_TOKEN_KEY)).getString();
86              idTokenJwt = ((JsonString) tokens.get("id_token_hint")).getString();
87              idToken = new JsonWebTokenProcessor(idTokenJwt).signatureCheck(false).getJsonPayload();
88              if (cookieValues.length == 1) {
89                  userInfo = null;
90              } else {
91                  userInfo = Json.createReader(CipherUtil.buildDecryptStream(new ByteArrayInputStream(Encoding.base64urlDecode(cookieValues[1])), secret))
92                          .readObject();
93              }
94          } catch (final IOException e) {
95              throw new GeneralSecurityException(e);
96          }
97      }
98  
99      /**
100      * Constructs with the ID token and user info.
101      *
102      * @param idToken
103      *            ID token
104      * @param userInfo
105      *            user info
106      */
107     public TokenCookie(final String accessToken, final String refreshToken, final JsonObject idToken, final String idTokenJwt, final JsonObject userInfo) {
108 
109         this.accessToken = accessToken;
110         this.refreshToken = refreshToken != null ? refreshToken : "";
111         this.idToken = idToken;
112         this.idTokenJwt = idTokenJwt;
113         this.userInfo = userInfo;
114     }
115 
116     /**
117      * Encodes the JSON object to a Base64 string.
118      *
119      * @param jsonObject
120      *            JSON obejct to encode
121      * @param secret
122      *            secret key
123      * @return encoded JSON object.
124      * @throws GeneralSecurityException
125      */
126     private String encode(final JsonObject jsonObject,
127             final SecretKey secret) throws GeneralSecurityException {
128 
129         try {
130             return Encoding.base64urlEncode(CipherUtil.encrypt(jsonObject.toString()
131                     .getBytes("UTF-8"), secret));
132         } catch (final IOException e) {
133             throw new GeneralSecurityException(e);
134         }
135     }
136 
137     public String getAccessToken() {
138 
139         return accessToken;
140     }
141 
142     public JsonObject getIdToken() {
143 
144         return idToken;
145     }

147     /**
148      * Returns the maximum age of the token.
149      *
150      * @return maximum age of the token
151      */
152     public int getMaxAge() {
153 
154         return idToken.getInt("exp") - (int) (System.currentTimeMillis() / 1000);
155     }
156 
157     public String getRefreshToken() {
158 
159         return refreshToken;
160     }
161 
162     public JsonObject getUserInfo() {
163 
164         return userInfo;
165     }
166 
167     public boolean isExpired() {
168 
169         return idToken.getInt("exp") < System.currentTimeMillis() / 1000;
170     }
171 
172     /**
173      * Converts to a cookie value.
174      *
175      * @param clientId
176      *            client ID (used for creating the {@link SecretKey})
177      * @param clientSecret
178      *            client secret (used for creating the {@link SecretKey})
179      * @return cookie value
180      * @throws GeneralSecurityException
181      */
182     public String toCookieValue(final SecretKey secret) throws GeneralSecurityException {
183 
184         final JsonObject tokens = Json.createObjectBuilder()
185                 .add(ACCESS_TOKEN_KEY, accessToken)
186                 .add(REFRESH_TOKEN_KEY, refreshToken)
187                 .add("id_token_hint", idTokenJwt)
188                 .build();
189 
190         final StringBuilder b = new StringBuilder(encode(tokens, secret));
191         if (userInfo != null) {
192             b.append('.');
193             b.append(encode(userInfo, secret));
194         }
195         return b.toString();
196     }
197 
198     public String getIdTokenJwt() {
199 
200         return idTokenJwt;
201     }
