20  /**
21   * This provides the base implementation for the OpenID JASPIC Module
22   * initializer. It is intended that applications will extend this and register
23   * themselves as a WebListener. This initializes the OpenID Connector JASPIC
24   * module and registers itself as the OAuth provider.
25   */
26  @WebListener
27  @Stateless
28  public class Initializer implements
29      ServletContextListener {
30  
31      @EJB
32      private OpenIdConnectModuleConfigProviderRemote config;
33  
34      /**
35       * A String identifier assigned by the {@link AuthConfigFactory} to the
36       * provider registration, and is used to remove the registration from the
37       * factory.
38       */
39      private String registrationID;
40  
41      @Override
42      public void contextDestroyed(final ServletContextEvent sce) {
43  
44          AuthConfigFactory.getFactory()
45              .removeRegistration(registrationID);
46      }
47  
48      @Override
49      public void contextInitialized(final ServletContextEvent sce) {
50  
51          final Map<String, String> options = new HashMap<>();
52  
53          options.put(CLIENT_ID, config.getClientId());
54          options.put(CLIENT_SECRET, config.getClientSecret());
55          options.put(OpenIdConnectAuthModule.DISABLE_CERTIFICATE_CHECKS_KEY, String.valueOf(config.isCertificateCheckDisabled()));
56          options.put(OpenIdConnectAuthModule.ISSUER_URI_KEY, config.getIssuerUri().toASCIIString());
57  
58          final String contextPath = sce.getServletContext()
59              .getContextPath();
60          final String rootPath = contextPath + "/";
61          options.put("cookie_context", rootPath);
62          options.put("scope", config.getScope());
63          options.put("redirection_endpoint", contextPath + "/cb");
64          options.put("logout_redirection_endpoint", contextPath + "/cblogout");
65          options.put("token_uri", contextPath + "/token");
66          options.put("userinfo_uri", contextPath + "/userinfo");
67          // TODO make this dependent on openid configuration
68          // options.put(OpenIdConnectAuthModule.LOGOUT_GOTO_URI_KEY, rootPath);
69          options.put(OpenIdConnectAuthModule.LOGOUT_URI_KEY, contextPath + "/logout");
70  
71          registrationID = AuthConfigFactory.getFactory()
72              .registerConfigProvider(new OpenIdConnectModuleConfigProvider(options, null), "HttpServlet", getAppContext(sce), null);
73          Log.fine("registered", registrationID, options);
74      }
75  
76      /**
77       * <p>
78       * The application context identifier (that is, the appContext parameter
79       * value) used to select the AuthConfigProvider and ServerAuthConfig objects
80       * for a specific application shall be the String value constructed by
81       * concatenating the host name, a blank separator character, and the decoded
82       * context path corresponding to the web module.
83       * </p>
84       *
85       * <pre>
86       * AppContextID ::= hostname blank context-path
87       * For example: "java-server /petstore"
88       * </pre>
89       * <p>
90       * This profile uses the term host name to refer to the name of a logical
91       * host that processes Servlet requests. Servlet requests may be directed to
92       * a logical host using various physical or virtual host names or addresses,
93       * and a message processing runtime may be composed of multiple logical
94       * hosts . Systems or administrators that register AuthConfigProvider object
95       * s with specific application context identifiers must have an ability to
96       * determine the host name for which they wish to perform the registration.
97       * </p>
98       *
99       * @see <a href=
100      *      "http://download.oracle.com/otn-pub/jcp/jaspic-1.0-fr-oth-JSpec/jaspic-1_0-fr-spec.pdf">
101      *      JASPIC Spec section 3.2</a>
102      * @return
103      */
104     private String getAppContext(final ServletContextEvent sce) {
105 
106         return sce.getServletContext()
107             .getVirtualServerName() + " "
108             + sce.getServletContext()
109                 .getContextPath();
110     }
111 }
