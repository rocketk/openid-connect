28      public static Map<String, Key> build(final JsonWebKeySet jwks) throws GeneralSecurityException {
29  
30          final Map<String, Key> keyMap = new HashMap<>();
31          for (final JsonWebKey key : jwks.getKeys()) {
32              keyMap.put(key.getKid(), key.toJcaKey());
33          }
34          return keyMap;
35      }
36  
37      public static Map<String, Key> build(final JsonObject jwks) throws GeneralSecurityException {
38  
39          final Map<String, Key> keyMap = new HashMap<>();
40          for (final JsonValue key : jwks.getJsonArray("keys")) {
41              final JsonObject keyObject = (JsonObject) key;
42              final String kid = keyObject.getString("kid");
43              final String kty = keyObject.getString("kty");
44              if ("RSA".equals(kty)) {
45                  keyMap.put(kid, buildRsaKey(keyObject));
46              } else if ("EC".equals(kty)) {
47                  keyMap.put(kid, buildEcKey(keyObject));
48              } else if ("oct".equals(kty)) {
49                  keyMap.put(kid, buildOctKey(keyObject));
50              } else {
51                  throw new NoSuchAlgorithmException("kty of " + kty + " is not supported");
52              }
53          }
54          return keyMap;
55      }

69      private static Key buildRsaKey(final JsonObject keyObject) throws GeneralSecurityException {
70  
71          if ("sig".equals(keyObject.getString("use"))) {
72              return buildRsaPublicKey(keyObject);
73          }
74          return null;
75      }
76  
77      private static Key buildRsaPublicKey(final JsonObject keyObject) throws GeneralSecurityException {
78  
79          final BigInteger modulus = Encoding.base64urlDecodeUint(keyObject.getString("n"));
80          final BigInteger publicExponent = Encoding.base64urlDecodeUint(keyObject.getString("e"));
81          return KeyFactory.getInstance("RSA")
82                  .generatePublic(new RSAPublicKeySpec(modulus, publicExponent));
83      }
